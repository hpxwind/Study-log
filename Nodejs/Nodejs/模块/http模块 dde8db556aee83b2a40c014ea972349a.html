http模块

用于创建http服务器和客户端，是构建web应用，api服务的基础。

  • 核心功能
    • 创建 HTTP 服务器：监听指定端口，处理客户端请求并返回响应。
    • 发送 HTTP 请求：作为客户端向其他服务器发起请求（GET、POST 等）。
    • 处理 HTTP 消息：解析请求头、请求体、构造响应等。
  • 基础示例与对象解析
    var http = require("http")

// 创建服务器
http.createServer((req, res) => {
    // 接收浏览器传的参数，返回渲染的内容
    // req 接收的参数
    // res 返回的渲染内容
    if (req.url == "/favicon.ico") {
        // 读取本地图标
        return
    }
    console.log(req.url)
    // 设置响应状态码和响应头（必须在发送响应体之前设置）
    res.writeHead(renderStatus(), {"content-type":"text/html;charset=utf-8"})
    // 响应内容，即显示页面
    res.write(renderHTML(req.url))
    // 发送响应体
    res.end()
})
// 监听端口3000和主机（默认0.0.0.0）
.listen(3000, ()=> {
    console.log("服务已启动，请访问：http://localhost:3000")
})

  • req与res对象解析

    req 代表客户端请求，主要属性和方法：

    method： 请求方法（'GET'、'POST' 等）
url：    请求 URL 路径（如 '/api/user'）
headers：请求头对象（如 { 'user-agent': 'curl/7.68.0', ... }）
httpVersion：          HTTP 版本（如 '1.1'）
on('data', callback)： 监听请求体数据（POST 等有请求体的方法）
on('end', callback)：  请求体接收完毕时触发
    示例：获取请求信息
server.on('request', (req, res) => {
  console.log('请求方法：', req.method);
  console.log('请求路径：', req.url);
  console.log('请求头：', req.headers);
  res.end('已收到请求');
});

    res 用于构造响应，主要方法：

    writeHead(statusCode, [headers])：设置响应状态码和响应头
statusCode：  HTTP 状态码（如 200 成功、404 未找到、500 服务器错误）
headers：     响应头对象（如 { 'Content-Type': 'application/json' }）
write(chunk)：发送响应体的一部分（可多次调用）
end([chunk])：结束响应（必须调用，否则客户端会一直等待）
setHeader(name, value)：单独设置响应头（需在 writeHead 或 end 前调用）
    示例：返回 JSON 响应
server.on('request', (req, res) => {
  res.writeHead(200, { 'Content-Type': 'application/json; charset=utf-8' });
  const data = { name: 'Node.js', version: 'v18.x' };
  res.end(JSON.stringify(data)); // 发送 JSON 字符串
});
  • JSONP

    JSONP 是一种早期时代的跨域请求方案，利用 <script> 标签不受不受同源策略限制的特性，通过动态标签加载远程脚本，实现跨域数据获取。

    原理：前端定义回调函数，后端返回「函数调用 + 数据」的脚本，前端执行时触发回调。

    对于接口统一，是前后端两情乡愿的事情，需要提前统一好。

    对于一个html文件（模拟跨域网站）：

    <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <script>
        var oscript = document.createElement("script")
        // 接收 http://localhost:3000/?callback=a001s11 并且协定好callback函数名字为s001s11
        oscript.src = `http://localhost:3000/?callback=a001s11`
				
        document.body.appendChild(oscript)
				
				// 用来接收函数，和提前协定好的函数名一样
        function a001s11(obj) {
            console.log("001s11", obj)
        }
    </script>
</body>
</html>

    对应后端需要解析域名中回调函数名

    var urlPath = url.parse(req.url, true)

    urlPath.query.callback 得到回调函数名，在res.end中添加返回即可

    var http = require("http")
var url = require("url")

http.createServer((req, res) => {
    var urlPath = url.parse(req.url, true)
    console.log(urlPath.query.callback)
    switch(urlPath.pathname) {
        case "/":
            res.end(`${urlPath.query.callback}(${JSON.stringify({
                "code": 200,
                "msg": "success",
                "data": "hello world"
            })})`)
            break
        case '/user':
            res.end(JSON.stringify({
                "code": 200,
                "msg": "success",
                "data": "hello user"
            }))
            break
        default:
            res.end(JSON.stringify({
                "code": 404,
                "msg": "not found"
            }))
            break
    }
}).listen(3000)
  • CORS

    在cors跨域问题中只需要在res.writeHead中添加"access-control-allow-origin": "*"，允许所有跨域请求即可

    res.writeHead(200,{
        "content-type":"application/json;charset=utf-8",

        // cors头
        "access-control-allow-origin": "*"
    })
