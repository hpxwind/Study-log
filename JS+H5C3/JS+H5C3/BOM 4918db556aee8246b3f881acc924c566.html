BOM

  • 基础认识

    浏览器对象模型，是一些操作浏览器的能力，比如

    获得一些浏览器的相关信息，

    浏览器跳转页面，

    地址栏信息，

    操作浏览器的滚动条，

    浏览器信息，

    让浏览器出现一个弹框。

    。。。

    BOM的核心是Window对象

    window是浏览器的一个内置的一个对象，里面包含着浏览器的操作方法。

  • 浏览器可视窗口的尺寸
    console.log(window.innerHeight, innerHeight); // 浏览器可视高度
console.log(window.innerWidth, innerWidth); // 浏览器可视宽度
  • 浏览器的弹出层

    1、弹出框alert()

    window.alert("jeo");
<button id="btn">click</button>

btn.onclick = function() {
    alert("hi");
}

    2、询问框confirm()

    <button id="btn">click</button>

btn.onclick = function() {
		var res = confirm("请输入:");
		console.log(res); // 返回值点击确定为true，否则为false
}

    3、输入框prompt()

    <button id="btn">click</button>

btn.onclick = function() {
    var res = prompt("请输入");
    console.log(res); // 返回输入值，不输入返回空
}
  • 浏览器地址栏

    1、window.location

    location.href = "https://www.baidu.com"

    location.reload()

    console.log(window.location);
console.log(location.herf) // 获取当前地址

    3、window.close()

    关掉自己的标签页

    // window.close()
btn.onclick  = function () {

  • 浏览器的常见事件

    1、window.onresize

    没每次改变可视窗口时执行

    window.onresize = function () {
		console.log("resized"); // 没每次改变可视窗口时会打印一次reiszed
}

    2、window.onload

    当页面中所有资源加载完毕后执行

    <button id="btn">click</button>

window.onload = function() {
		console.log("loaded up");
		console.log(btn);
}

    3、window.onscroll & document.documentElement.scrollTop & window.scrollTo()

    页面滚动时执行

    document.documentElement.scrollTop

    document.documentElement.scrollLeft

    计算距离顶部的高度/左侧的宽度

    如果没用<!DOCTYPE html>（低版本浏览器）时，使用

    document.body.scrollTop

    兼容性解决

    document.documentElement.scrollTop || document.body.scrollTop

    window.scrollTo(0,0) 滚动到(x,y)位置

    window.onscroll = function(){
		console.log("scroll", "滚动距离");
}

window.onscroll = function() {
    console.log(document.documentElement.scrollTop || document.body.scrollTop);// 兼容所有浏览器版本
}

window.onscroll = function() {
    if ((document.documentElement.scrollTop || document.body.scrollTop) > 150) {
        console.log("显示");
    } else {
        console.log("隐藏");
    }
}

btn.onclick = function() {
    window.scrollTo(0,0); // x滚动到0， y滚动到0
}
  • 浏览器历史记录操作

    1、history.back() & history.forward()

    返回最近一次前进跳转，前提是有跳转

    前进最近一次后退跳转，前提是有回退

    btn.onclick = function(){
    // location.href = "数字操作.html";
    history.forward(); // history.go(-1);后退，传1就是前进
}
    btn.onclick = function() {
    history.back();
}

    2、history.go()

    history.go(Number)

    Number > 0 前进 Number个页面

    Number < 0 后退 Number个页面

  • 浏览器本地存储

    localStorage为永久存储

    sessionStorage为临时存储，关闭页面就丢失了

    1、localStorage

    // 增
localStorage.setItem("name", "k");
// 取
localStorage.getItem("name");
// 删
localStorage.removeItem("name");
// 清空
localStorage.clear()

    2、sessionStorage

    // 增
sessionStorage.setItem("name", "K");
// 取
sessionStorage.getItem("name");
// 删
sessionStorage.removeItem("name");
// 清空
sessionStorage.clear()
    <button id="btn">存</button>
<button id="btn2">取</button>
<button id="btn3">删</button>
<button id="btn4">clear</button>
<script>
    btn.onclick = function() { // 只能存字符串
        localStorage.setItem("age", "200");
        localStorage.setItem("obj", JSON.stringify({name:"k", age:100}));
    }

    btn2.onclick = function() {
        console.log(localStorage.getItem("age"));
        console.log(JSON.parse(localStorage.getItem("obj")));
    }

    btn3.onclick = function() {
        localStorage.removeItem("age");
    }

    btn4.onclick = function() {
        localStorage.clear();
    }
</script>
  • 记住用户名案例
    <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <div>
        username:
        <input type="text" id="username">
    </div>

    <div>
        password:
        <input type="password" id="password">
    </div>

    <div>
        <button id="login"> login</button>
    </div>

    <script>
        // 先获取用户名信息
        var getUsername = localStorage.getItem("username");
        var getPassword = localStorage.getItem("password");

        if (getUsername && getPassword) {
            username.value = getUsername;
            password.value = getPassword;
        }

        login.onclick = function() {
            // console.log(username.value, password.value);
            localStorage.setItem("username", username.value);
            localStorage.setItem("password", password.value);
        }
    </script>
</body>
</html>
