ArrayList

核心概念

// 传统数组的局限性
int[] fixedArray = new int[10];  // 固定大小，无法扩容
fixedArray[10] = 11;  // ❌ ArrayIndexOutOfBoundsException

// 动态数组的优势
List<Integer> dynamicList = new ArrayList<>();
dynamicList.add(1);  // 自动扩容
dynamicList.add(2);
// ✅ 无需预先知道大小

创建和初始化

// 创建空列表
List<String> list1 = new ArrayList<>();

// 指定初始容量
List<String> list2 = new ArrayList<>(20);

// 从其他集合创建
List<String> list3 = new ArrayList<>(Arrays.asList("A", "B", "C"));

// 使用静态工厂方法
List<String> list4 = List.of("Java", "Python", "C++");  // 不可变
List<String> list5 = new ArrayList<>(List.of("A", "B"));  // 可变

添加元素

List<Integer> list = new ArrayList<>();

// 尾部添加
boolean added = list.add(1);       // 总是返回true
list.add(2);
list.add(3);

// 指定位置插入
list.add(0, 0);      // 头部插入
list.add(2, 100);    // 索引2处插入
list.add(list.size(), 999);  // 等效于尾部添加

// 批量添加
list.addAll(Arrays.asList(4, 5, 6));
list.addAll(1, Arrays.asList(7, 8, 9));  // 指定位置批量添加

获取元素

List<String> list = new ArrayList<>(Arrays.asList("A", "B", "C", "D"));

// 获取单个元素
String first = list.get(0);   // "A"
String last = list.get(list.size() - 1);  // "D"

// 遍历
for (int i = 0; i < list.size(); i++) {
    System.out.println(list.get(i));
}

// 增强for循环
for (String item : list) {
    System.out.println(item);
}

// 使用迭代器
Iterator<String> it = list.iterator();
while (it.hasNext()) {
    System.out.println(it.next());
}

// Lambda表达式
list.forEach(item -> System.out.println(item));
list.forEach(System.out::println);

修改元素

List<String> list = new ArrayList<>(Arrays.asList("A", "B", "C"));

// 替换元素
String oldValue = list.set(1, "X");  // 返回"B"，列表变为["A", "X", "C"]

// 批量替换
list.replaceAll(s -> s + "-1");  // 所有元素加"-1"，变成["A-1", "X-1", "C-1"]

删除元素

List<Integer> list = new ArrayList<>(Arrays.asList(1, 2, 3, 4, 5, 3));

// 删除指定索引的元素
int removed = list.remove(1);  // 删除索引1，返回2

// 删除第一个匹配的元素
boolean removed2 = list.remove(Integer.valueOf(3));  // 删除第一个3，返回true
boolean removed3 = list.remove(Integer.valueOf(10)); // 返回false

// 删除多个元素
list.removeAll(Arrays.asList(3, 4));  // 删除所有3和4
list.retainAll(Arrays.asList(1, 5));  // 只保留1和5，删除其他

// 条件删除
list.removeIf(n -> n > 3);  // 删除所有大于3的元素

// 清空
list.clear();  // 清空所有

int firstIndex = list.indexOf("A");     // 0

批量操作

List<Integer> list1 = new ArrayList<>(Arrays.asList(1, 2, 3));
List<Integer> list2 = new ArrayList<>(Arrays.asList(2, 3, 4));

// 交集
list1.retainAll(list2);  // list1变为[2, 3]

// 并集
list1.addAll(list2);  // 将list2所有元素加到list1末尾

// 差集
list1.removeAll(list2);  // 从list1中删除list2中所有元素

// 相等判断
boolean equal = list1.equals(Arrays.asList(1, 2, 3));  // 比较内容

排序和比较

List<Integer> numbers = new ArrayList<>(Arrays.asList(5, 2, 8, 1, 9));

// 自然排序
Collections.sort(numbers);  // 升序
numbers.sort(null);         // 同上
numbers.sort(Comparator.naturalOrder());  // 升序
numbers.sort(Comparator.reverseOrder());  // 降序

// 自定义排序
numbers.sort((a, b) -> b - a);  // 降序
numbers.sort(Comparator.comparingInt(n -> n));  // 升序
numbers.sort(Comparator.comparingInt(n -> n).reversed());  // 降序

// 复杂对象排序
List<Person> people = new ArrayList<>();
people.sort(Comparator.comparing(Person::getAge).thenComparing(Person::getName));

// 随机排序
Collections.shuffle(numbers);

流操作

List<Integer> list = Arrays.asList(1, 2, 3, 4, 5);

// 过滤
List<Integer> filtered = list.stream()
    .filter(n -> n > 2)
    .collect(Collectors.toList());

// 映射
List<String> mapped = list.stream()
    .map(n -> "Number: " + n)
    .collect(Collectors.toList());

// 聚合
int sum = list.stream().mapToInt(Integer::intValue).sum();
Optional<Integer> max = list.stream().max(Integer::compare);
Optional<Integer> min = list.stream().min(Integer::compare);
double average = list.stream().mapToInt(Integer::intValue).average().orElse(0.0);

// 统计
IntSummaryStatistics stats = list.stream().mapToInt(Integer::intValue).summaryStatistics();
long count = stats.getCount();
int maxNum = stats.getMax();
int minNum = stats.getMin();
double avg = stats.getAverage();
long sumNum = stats.getSum();

// 去重
List<Integer> distinct = list.stream().distinct().collect(Collectors.toList());

// 收集
Set<Integer> set = list.stream().collect(Collectors.toSet());
Map<Integer, Long> countMap = list.stream()
    .collect(Collectors.groupingBy(n -> n, Collectors.counting()));

常见错误

// 错误：遍历时删除
List<String> list = new ArrayList<>(Arrays.asList("A", "B", "C"));
for (int i = 0; i < list.size(); i++) {
    if (list.get(i).equals("B")) {
        list.remove(i);  // 会跳过检查C
    }
}

// 正确：使用迭代器删除
Iterator<String> it = list.iterator();
while (it.hasNext()) {
    if (it.next().equals("B")) {
        it.remove();  // 安全删除
    }
}

// 正确：从后往前删除
for (int i = list.size() - 1; i >= 0; i--) {
    if (list.get(i).equals("B")) {
        list.remove(i);
    }
}
