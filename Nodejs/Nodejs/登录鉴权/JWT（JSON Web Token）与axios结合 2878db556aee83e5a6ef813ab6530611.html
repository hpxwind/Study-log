JWT（JSON Web Token）与axios结合

一、依赖安装

npm i express jsonwebtoken express cors

二、后端 JWT 签发 & 校验（server.js）

const express = require('express')
const jwt = require('jsonwebtoken')
const cors = require('cors')
const app = express()

app.use(cors())
app.use(express.json())

// 密钥（项目放环境变量）
const SECRET_KEY = 'my_jwt_secret_2026'
// 过期时间 2小时
const EXPIRE = '2h'

// 1.登录接口：签发JWT
app.post('/login',(req,res)=>{
  const {username,pwd} = req.body
  // 模拟校验账号
  if(username==='admin'&&pwd==='123456'){
    // 生成token
    const token = jwt.sign(
      {id:1,username},
      SECRET_KEY,
      {expiresIn:EXPIRE}
    )
    res.json({code:200,token,msg:'登录成功'})
  }else{
    res.json({code:400,msg:'账号密码错误'})
  }
})

// 2.JWT校验中间件
function authToken(req,res,next){
  const auth = req.headers.authorization
  if(!auth) return res.json({code:401,msg:'无令牌，请登录'})
  const token = auth.split(' ')[1]
  try{
    const user = jwt.verify(token,SECRET_KEY)
    req.user = user
    next()
  }catch(err){
    res.json({code:401,msg:'令牌失效/过期'})
  }
}

// 3.需要权限的接口
app.get('/user/info',authToken,(req,res)=>{
  res.json({code:200,data:req.user})
})

app.listen(3000,()=>console.log('服务启动 3000'))

三、前端 Axios 封装 + 携带 Token

1. 安装 axios

npm i axios

2. axios 请求封装（api.js）

import axios from 'axios'

const request = axios.create({
  baseURL:'http://localhost:3000',
  timeout:5000
})

// 请求拦截器：自动带上Token
request.interceptors.request.use(config=>{
  const token = localStorage.getItem('token')
  if(token){
    // Bearer 固定格式
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// 响应拦截器：统一处理过期、错误
request.interceptors.response.use(
  res=>res.data,
  err=>{
    const res = err.response.data
    if(res.code===401){
      localStorage.removeItem('token')
      alert('登录失效，请重新登录')
      // 跳转登录页
      // router.push('/login')
    }
    return Promise.reject(err)
  }
)

export default request

四、页面调用示例

import request from './api'

// 登录，保存token
async function login(){
  let res = await request.post('/login',{
    username:'admin',
    pwd:'123456'
  })
  if(res.code===200){
    localStorage.setItem('token',res.token)
    alert('登录成功')
  }
}

// 携带token请求个人信息
async function getUserInfo(){
  let data = await request.get('/user/info')
  console.log('用户信息',data)
}

五、核心关键点

  1. Token 存放：前端存在localStorage
  1. 请求头格式：Authorization: Bearer 令牌
  1. 拦截器作用：统一加 token、统一鉴权报错
  1. 过期逻辑：后端校验失败 → 前端清空 token 跳转登录
  1. 退出登录：直接删除本地token即可

六、常见问题

  • 跨域：后端开启cors即可
  • 401 报错：token 丢失、过期、密钥不一致
  • 请求没带 token：检查请求拦截器是否生效
