      package com.hpxwind.tutorial.item.custom;

import net.minecraft.network.chat.Component;
import net.minecraft.world.InteractionResult;
import net.minecraft.world.entity.player.Player;
import net.minecraft.world.item.Item;
import net.minecraft.world.item.context.UseOnContext;
import net.minecraft.world.level.Level;
import net.minecraft.world.level.block.Blocks;
import net.minecraft.world.level.block.state.BlockState;

public class ProspectorItem extends Item {
    public ProspectorItem(Properties pProperties) {
        super(pProperties);
    }

    @Override
    public InteractionResult useOn(UseOnContext context) {
        var blockPos = context.getClickedPos();
        Player player = context.getPlayer();
        Level world = context.getLevel();

        // 只在服务端执行
        if (!world.isClientSide()) {
            boolean found = false;

            if (!player.isCrouching()) {
                // 正常使用：探测 5x5x64 大范围
                for (int y = 0; y <= 64; y++) { // 向下探测64格
                    for (int x = -2; x <= 2; x++) { // 左右3格
                        for (int z = -2; z <= 2; z++) { // 前后3格
                            var pos = blockPos.offset(x, -y, z); // ✅ 正确坐标
                            BlockState state = world.getBlockState(pos);

                            if (isCorrectBlock(state)) {
                                player.sendSystemMessage(Component.literal("Found " + state.getBlock().getName().getString() + "!"));
                                found = true;
                                break;
                            }
                        }
                        if (found) break;
                    }
                    if (found) break;
                }
            } else {
                // 蹲下使用：垂直向下探测
                for (int y = 0; y <= 64; y++) {
                    var pos = blockPos.below(y);
                    BlockState state = world.getBlockState(pos);

                    if (isCorrectBlock(state)) {
                        player.sendSystemMessage(Component.literal("Found " + state.getBlock().getName().getString() + "!"));
                        found = true;
                        break;
                    }
                }
            }

            if (!found) {
                player.sendSystemMessage(Component.literal("No ore found!"));
            }

            // 消耗耐久
            context.getItemInHand().hurtAndBreak(1, player, p -> p.broadcastBreakEvent(context.getHand()));
            return InteractionResult.SUCCESS;
        }
        return super.useOn(context);
    }

    // 探测目标：钻石块 + 深层钻石矿石
    private boolean isCorrectBlock(BlockState blockState) {
        return blockState.is(Blocks.DIAMOND_BLOCK)
                || blockState.is(Blocks.DEEPSLATE_DIAMOND_ORE)
                || blockState.is(Blocks.DIAMOND_ORE); // 原版钻石矿石也加上
    }
}

勘探杖核心方法

  • 方法汇总

    1. useOn(UseOnContext context)

    • 作用：右键点击方块时触发
    • 用途：写工具右键功能（探测、放置、改变方块等）
    • 必须重写才能实现自定义工具效果

    2. BlockPos

    • 作用：存储方块的 X/Y/Z 三维坐标
    • 理解：MC 世界里每个方块都有坐标，这个类就是定位用的
    • 常用获取：context.getClickedPos() → 获取右键点的方块位置

    3. blockPos.offset(x, y, z)

    • 作用：在原坐标基础上偏移
    • 用途：范围搜索、前后左右查找方块
    • 例子：
      • offset(1,0,0) → 右边 1 格
      • offset(0,-1,0) → 下面 1 格
      • offset(-2,0,3) → 左 2 格，后 3 格

    4. world.getBlockState(pos)

    • 作用：获取某个坐标的方块是什么
    • 返回：BlockState（方块状态）
    • 用途：判断这个位置是不是矿石、石头、钻石等

    5. !world.isClientSide()

    • 作用：判断当前是否在 服务端 运行
    • 规则：
      • 客户端 → 只负责显示画面
      • 服务端 → 负责所有逻辑（判断、搜索、消耗耐久）
    • 必须写，不写会出现：卡顿、崩溃、提示重复、功能失效

    6. item.hurtAndBreak(消耗值, 玩家, 事件)

    • 作用：工具消耗耐久
    • 标准写法：
      context.getItemInHand().hurtAndBreak(1, player,
   p -> p.broadcastBreakEvent(context.getHand()));
    • 功能：每次使用 -1 耐久，耐久为 0 工具破碎
