  • 哈希（hash）计算

    哈希函数将任意长度的数据转换为固定长度的哈希值（摘要），具有不可逆性，常用于密码存储、数据校验。

    常用算法：md5（不推荐，安全性低）、sha1、sha256、sha512 等。

    示例：计算数据的 SHA256 哈希

    function calculateHash(data) {
  // 创建哈希对象（指定算法）
  const hash = crypto.createHash('sha256');
  
  // 更新要计算哈希的数据（可多次调用 update）
  hash.update(data);
  
  // 生成哈希值（指定输出格式：hex/base64等）
  return hash.digest('hex');
}

const data = 'hello crypto';
const hashValue = calculateHash(data);
console.log('SHA256 哈希值：', hashValue);
// 输出类似：a26b18...（64位十六进制字符串）

    密码存储最佳实践：哈希时添加盐值（salt）防止彩虹表攻击

    function hashPassword(password) {
  // 生成随机盐值
  const salt = crypto.randomBytes(16).toString('hex');
  
  // 结合盐值计算哈希
  const hash = crypto.createHash('sha256');
  hash.update(password + salt);
  const hashedPassword = hash.digest('hex');
  
  // 存储盐值和哈希值（验证时需要盐值）
  return { salt, hashedPassword };
}

// 验证密码
function verifyPassword(password, storedSalt, storedHash) {
  const hash = crypto.createHash('sha256');
  hash.update(password + storedSalt);
  const inputHash = hash.digest('hex');
  return inputHash === storedHash;
}
  • 对称加密（AES）

    对称加密使用同一个密钥进行加密和解密，速度快，适合大量数据加密。

    常用模式：aes-128-cbc、aes-256-gcm（推荐，带认证）等。

    示例：AES-GCM 加密解密

    function encrypt(text, key) {
  // 生成随机初始化向量（IV）
  const iv = crypto.randomBytes(12); // GCM模式推荐12字节IV
  
  // 创建加密器
  const cipher = crypto.createCipheriv('aes-256-gcm', key, iv);
  
  // 加密数据
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  
  // 获取认证标签（GCM模式需要）
  const authTag = cipher.getAuthTag().toString('hex');
  
  return { iv: iv.toString('hex'), encryptedData: encrypted, authTag };
}

function decrypt(encryptedData, key, iv, authTag) {
  // 创建解密器
  const decipher = crypto.createDecipheriv(
    'aes-256-gcm',
    key,
    Buffer.from(iv, 'hex')
  );
  
  // 设置认证标签
  decipher.setAuthTag(Buffer.from(authTag, 'hex'));
  
  // 解密数据
  let decrypted = decipher.update(encryptedData, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  
  return decrypted;
}

// 使用示例
const key = crypto.scryptSync('my-secret-key', 'salt', 32); // 生成32字节密钥（AES-256）
const text = '需要加密的敏感数据';

const encrypted = encrypt(text, key);
console.log('加密后：', encrypted);

const decrypted = decrypt(encrypted.encryptedData, key, encrypted.iv, encrypted.authTag);
console.log('解密后：', decrypted); // 输出：需要加密的敏感数据

    注意：

    • 密钥长度需与算法匹配（如 aes-256 需要 32 字节密钥）
    • 每次加密应使用随机 IV（初始化向量）
    • GCM 模式提供认证功能，更安全
  • 非对称加密（RSA）

    非对称加密使用公钥加密、私钥解密（或私钥签名、公钥验证），适合密钥交换、数字签名。

    示例：RSA 密钥对生成与加密解密

    // 生成 RSA 密钥对
function generateKeyPair() {
  return crypto.generateKeyPairSync('rsa', {
    modulusLength: 2048, // 密钥长度
    publicKeyEncoding: {
      type: 'spki',
      format: 'pem'
    },
    privateKeyEncoding: {
      type: 'pkcs8',
      format: 'pem'
    }
  });
}

// 加密（使用公钥）
function rsaEncrypt(text, publicKey) {
  const buffer = Buffer.from(text, 'utf8');
  const encrypted = crypto.publicEncrypt(publicKey, buffer);
  return encrypted.toString('base64');
}

// 解密（使用私钥）
function rsaDecrypt(encryptedText, privateKey) {
  const buffer = Buffer.from(encryptedText, 'base64');
  const decrypted = crypto.privateDecrypt(privateKey, buffer);
  return decrypted.toString('utf8');
}

// 使用示例
const { publicKey, privateKey } = generateKeyPair();
const text = '通过RSA加密的信息';

const encrypted = rsaEncrypt(text, publicKey);
console.log('RSA加密后：', encrypted);

const decrypted = rsaDecrypt(encrypted, privateKey);
console.log('RSA解密后：', decrypted);
  • 数字签名

    使用私钥签名数据，公钥验证签名，确保数据完整性和来源真实性。

    示例：RSA 签名与验证

    // 签名（私钥）
function signData(data, privateKey) {
  const sign = crypto.createSign('sha256');
  sign.update(data);
  sign.end();
  return sign.sign(privateKey, 'base64'); // 返回签名
}

// 验证（公钥）
function verifySignature(data, signature, publicKey) {
  const verify = crypto.createVerify('sha256');
  verify.update(data);
  verify.end();
  return verify.verify(publicKey, signature, 'base64'); // 返回布尔值
}

// 使用示例
const { publicKey, privateKey } = generateKeyPair();
const data = '需要签名的数据';

const signature = signData(data, privateKey);
console.log('签名：', signature);

const isValid = verifySignature(data, signature, publicKey);
console.log('签名验证结果：', isValid); // 输出：true
  • 随机数生成

    生成加密安全的随机数，用于密钥、盐值、IV 等场景。

    // 生成16字节随机数（128位）
const randomBytes = crypto.randomBytes(16);
console.log('随机字节：', randomBytes);
console.log('十六进制：', randomBytes.toString('hex'));
console.log('Base64：', randomBytes.toString('base64'));
  • 注意
    1. 算法选择：避免使用不安全的算法（如 md5、sha1、aes-ecb）
    1. 密钥管理：密钥需安全存储，避免硬编码在代码中
    1. 性能考量：非对称加密速度较慢，适合小数据（如密钥交换），大量数据优先用对称加密
    1. 错误处理：加密操作可能抛出错误（如密钥长度不匹配），需使用 try/catch 处理
