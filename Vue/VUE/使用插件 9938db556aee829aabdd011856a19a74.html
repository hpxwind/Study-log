使用插件

插件的下载到使用(以Swiper为例子)：

  1. 插件安装
# npm安装
npm install swiper

# yarn安装
yarn add swiper

# 如果需要特定版本
npm install swiper@8.4.5
  1. 完整引入（传统方式）
// main.js
import Vue from 'vue'
import App from './App.vue'

// 引入Swiper样式
import 'swiper/swiper-bundle.css'

// 引入Swiper核心和所需模块
import Swiper, { Navigation, Pagination, Autoplay } from 'swiper'

// 配置Swiper模块
Swiper.use([Navigation, Pagination, Autoplay])

new Vue({
  render: h => h(App),
}).$mount('#app')
  1. 按需引入（推荐）
// main.js
import Vue from 'vue'
import App from './App.vue'

// 只引入需要的模块和样式
import { Swiper, Navigation, Pagination, Autoplay } from 'swiper'
import 'swiper/css'
import 'swiper/css/navigation'
import 'swiper/css/pagination'

// 注册模块
Swiper.use([Navigation, Pagination, Autoplay])

new Vue({
  render: h => h(App),
}).$mount('#app')

在Vue组件中使用Swiper

  • 选项式API使用方式 (vue2)
    <template>
  <div class="swiper-container">
    <div class="swiper-wrapper">
      <div 
        class="swiper-slide" 
        v-for="(slide, index) in slides" 
        :key="index"
      >
        <img :src="slide.image" :alt="slide.title">
        <div class="slide-content">
          <h3>{{ slide.title }}</h3>
          <p>{{ slide.description }}</p>
        </div>
      </div>
    </div>
    
    <!-- 导航按钮 -->
    <div class="swiper-button-next"></div>
    <div class="swiper-button-prev"></div>
    
    <!-- 分页器 -->
    <div class="swiper-pagination"></div>
  </div>
</template>

<script>
export default {
  name: 'SwiperDemo',
  data() {
    return {
      slides: [
        {
          image: 'https://via.placeholder.com/800x400/FF6B6B/FFFFFF?text=Slide+1',
          title: '第一张幻灯片',
          description: '这是第一张幻灯片的描述内容'
        },
        {
          image: 'https://via.placeholder.com/800x400/4ECDC4/FFFFFF?text=Slide+2',
          title: '第二张幻灯片',
          description: '这是第二张幻灯片的描述内容'
        },
        {
          image: 'https://via.placeholder.com/800x400/45B7D1/FFFFFF?text=Slide+3',
          title: '第三张幻灯片',
          description: '这是第三张幻灯片的描述内容'
        }
      ],
      swiperInstance: null
    }
  },
  mounted() {
    this.initSwiper()
  },
  beforeDestroy() {
    // 销毁Swiper实例，避免内存泄漏
    if (this.swiperInstance) {
      this.swiperInstance.destroy(true, true)
    }
  },
  methods: {
    initSwiper() {
      this.$nextTick(() => {
        this.swiperInstance = new Swiper('.swiper-container', {
          // 基本配置
          loop: true,
          speed: 600,
          
          // 自动播放
          autoplay: {
            delay: 3000,
            disableOnInteraction: false,
          },
          
          // 导航按钮
          navigation: {
            nextEl: '.swiper-button-next',
            prevEl: '.swiper-button-prev',
          },
          
          // 分页器
          pagination: {
            el: '.swiper-pagination',
            clickable: true,
            dynamicBullets: true,
          },
          
          // 响应式断点
          breakpoints: {
            640: {
              slidesPerView: 1,
              spaceBetween: 20,
            },
            768: {
              slidesPerView: 2,
              spaceBetween: 30,
            },
            1024: {
              slidesPerView: 3,
              spaceBetween: 40,
            },
          },
          
          // 事件监听
          on: {
            slideChange: () => {
              console.log('当前slide索引:', this.swiperInstance.activeIndex)
            },
            reachEnd: () => {
              console.log('到达最后一张')
            }
          }
        })
      })
    },
    
    // 控制方法
    goToSlide(index) {
      if (this.swiperInstance) {
        this.swiperInstance.slideTo(index)
      }
    },
    
    nextSlide() {
      if (this.swiperInstance) {
        this.swiperInstance.slideNext()
      }
    },
    
    prevSlide() {
      if (this.swiperInstance) {
        this.swiperInstance.slidePrev()
      }
    },
    
    startAutoplay() {
      if (this.swiperInstance) {
        this.swiperInstance.autoplay.start()
      }
    },
    
    stopAutoplay() {
      if (this.swiperInstance) {
        this.swiperInstance.autoplay.stop()
      }
    }
  }
}
</script>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue'
import { Swiper, Navigation, Pagination, Autoplay } from 'swiper'
import 'swiper/css'
import 'swiper/css/navigation'
import 'swiper/css/pagination'

// Swiper容器引用
const swiperContainer = ref(null)
const swiperInstance = ref(null)
const isAutoplay = ref(true)

// 幻灯片数据
const slides = ref([
  {
    image: 'https://via.placeholder.com/800x400/FF6B6B/FFFFFF?text=Vue+3+Slide+1',
    title: 'Vue 3 集成 Swiper',
    description: '展示如何在Vue 3项目中使用Swiper轮播图'
  },
  {
    image: 'https://via.placeholder.com/800x400/4ECDC4/FFFFFF?text=Vue+3+Slide+2',
    title: '组合式API',
    description: '使用Vue 3的组合式API管理Swiper实例'
  },
  {
    image: 'https://via.placeholder.com/800x400/45B7D1/FFFFFF?text=Vue+3+Slide+3',
    title: '响应式设计',
    description: 'Swiper支持响应式布局和触摸操作'
  }
])

// 初始化Swiper
const initSwiper = () => {
  if (!swiperContainer.value) return
  
  swiperInstance.value = new Swiper(swiperContainer.value, {
    loop: true,
    speed: 600,
    autoplay: {
      delay: 3000,
      disableOnInteraction: false,
    },
    navigation: {
      nextEl: '.swiper-button-next',
      prevEl: '.swiper-button-prev',
    },
    pagination: {
      el: '.swiper-pagination',
      clickable: true,
    },
    on: {
      slideChange: () => {
        console.log('当前slide索引:', swiperInstance.value.activeIndex)
      }
    }
  })
  
  // 监听autoplay状态
  swiperInstance.value.on('autoplayStart', () => {
    isAutoplay.value = true
  })
  
  swiperInstance.value.on('autoplayStop', () => {
    isAutoplay.value = false
  })
}

// 控制方法
const goToSlide = (index) => {
  if (swiperInstance.value) {
    swiperInstance.value.slideTo(index)
  }
}

const nextSlide = () => {
  if (swiperInstance.value) {
    swiperInstance.value.slideNext()
  }
}

const prevSlide = () => {
  if (swiperInstance.value) {
    swiperInstance.value.slidePrev()
  }
}

const toggleAutoplay = () => {
  if (!swiperInstance.value) return
  
  if (isAutoplay.value) {
    swiperInstance.value.autoplay.stop()
  } else {
    swiperInstance.value.autoplay.start()
  }
}

// 生命周期
onMounted(() => {
  initSwiper()
})

onUnmounted(() => {
  if (swiperInstance.value) {
    swiperInstance.value.destroy(true, true)
  }
})
</script>

封装Swiper组件

  • 基础封装组件
    <!-- components/SwiperWrapper.vue -->
<template>
  <div :class="['swiper-wrapper-component', customClass]" :style="customStyle">
    <div ref="swiperEl" class="swiper">
      <div class="swiper-wrapper">
        <slot></slot>
      </div>
      
      <!-- 导航按钮 -->
      <div v-if="navigation" class="swiper-button-next"></div>
      <div v-if="navigation" class="swiper-button-prev"></div>
      
      <!-- 分页器 -->
      <div v-if="pagination" class="swiper-pagination"></div>
      
      <!-- 滚动条 -->
      <div v-if="scrollbar" class="swiper-scrollbar"></div>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted, onUnmounted, watch, nextTick } from 'vue'
import { Swiper } from 'swiper'

const props = defineProps({
  options: {
    type: Object,
    default: () => ({})
  },
  navigation: {
    type: Boolean,
    default: false
  },
  pagination: {
    type: Boolean,
    default: false
  },
  scrollbar: {
    type: Boolean,
    default: false
  },
  autoplay: {
    type: [Boolean, Object],
    default: false
  },
  loop: {
    type: Boolean,
    default: false
  },
  customClass: String,
  customStyle: Object
})

const emit = defineEmits(['swiper-init', 'slide-change'])

const swiperEl = ref(null)
const swiperInstance = ref(null)

// 默认配置
const defaultOptions = {
  loop: props.loop,
  speed: 600,
  autoplay: typeof props.autoplay === 'object' ? props.autoplay : (props.autoplay ? { delay: 3000 } : false),
  navigation: props.navigation ? {
    nextEl: '.swiper-button-next',
    prevEl: '.swiper-button-prev',
  } : false,
  pagination: props.pagination ? {
    el: '.swiper-pagination',
    clickable: true,
  } : false,
  scrollbar: props.scrollbar ? {
    el: '.swiper-scrollbar',
    hide: true,
  } : false,
  on: {
    slideChange: function() {
      emit('slide-change', {
        activeIndex: this.activeIndex,
        realIndex: this.realIndex,
        swiper: this
      })
    }
  }
}

// 合并配置
const mergeOptions = () => {
  return {
    ...defaultOptions,
    ...props.options,
    autoplay: props.options.autoplay !== undefined ? props.options.autoplay : defaultOptions.autoplay,
    navigation: props.options.navigation !== undefined ? props.options.navigation : defaultOptions.navigation,
    pagination: props.options.pagination !== undefined ? props.options.pagination : defaultOptions.pagination,
    scrollbar: props.options.scrollbar !== undefined ? props.options.scrollbar : defaultOptions.scrollbar,
  }
}

// 初始化Swiper
const initSwiper = () => {
  if (!swiperEl.value) return
  
  nextTick(() => {
    swiperInstance.value = new Swiper(swiperEl.value, mergeOptions())
    emit('swiper-init', swiperInstance.value)
  })
}

// 更新Swiper
const updateSwiper = () => {
  if (swiperInstance.value) {
    swiperInstance.value.update()
  }
}

// 销毁Swiper
const destroySwiper = () => {
  if (swiperInstance.value) {
    swiperInstance.value.destroy(true, true)
    swiperInstance.value = null
  }
}

// 暴露方法给父组件
defineExpose({
  swiper: swiperInstance,
  update: updateSwiper,
  destroy: destroySwiper,
  slideTo: (index) => swiperInstance.value?.slideTo(index),
  slideNext: () => swiperInstance.value?.slideNext(),
  slidePrev: () => swiperInstance.value?.slidePrev(),
})

// 监听配置变化
watch(() => props.options, () => {
  destroySwiper()
  initSwiper()
}, { deep: true })

// 生命周期
onMounted(() => {
  initSwiper()
})

onUnmounted(() => {
  destroySwiper()
})
</script>

    <template>
  <div class="demo-page">

            },
          }

        </div>
      </SwiperWrapper>
    </section>
