字符串比较

String s1 = "Hello";
String s2 = "hello";
String s3 = "Hello";
String s4 = new String("Hello");

// 比较值
boolean b1 = s1.equals(s2);        // false
boolean b2 = s1.equalsIgnoreCase(s2); // true
boolean b3 = s1.equals(s3);        // true

// 比较引用
boolean b4 = s1 == s3;             // true (字符串池)
boolean b5 = s1 == s4;             // false (不同对象)

// 字典序比较
int cmp1 = s1.compareTo(s2);       // 负数 (H < h)
int cmp2 = s1.compareToIgnoreCase(s2); // 0

// 内容比较
boolean b6 = s1.contentEquals(s4); // true

字符串修改

String str = "  Hello Java  ";

// 去除空白
String trimmed = str.trim();                    // "Hello Java"

String replaced1 = str.replace('a', 'o');       // "  Hello Jovo  "

String replaced3 = str.replaceAll("\\s+", "-");  // "--Hello-Java--" (正则)

// 大小写转换
String upper = str.toUpperCase();  // "  HELLO JAVA  "

// 转义/反转
String reversed = new StringBuilder(str).reverse().toString();  // "  avaJ olleH  "

正则表达式操作

String str = "abc123def456";

// 匹配
boolean matches = str.matches(".*\\d+.*");  // true

// 替换
String replaced = str.replaceAll("\\d+", "#");  // "abc#def#"

// 分割
String[] parts = str.split("\\d+");  // ["abc", "def", ""]

// 查找所有匹配
Pattern pattern = Pattern.compile("\\d+");
Matcher matcher = pattern.matcher(str);
while (matcher.find()) {
    System.out.println(matcher.group());  // 123, 456
}

字符串池与intern()

String s1 = "Hello";  // 字符串池
String s2 = new String("Hello");  // 堆内存
String s3 = s2.intern();  // 返回字符串池中的引用

System.out.println(s1 == s2);  // false
System.out.println(s1 == s3);  // true

Java 11+ 新特性

// repeat() - 重复字符串
String repeated = "Hi".repeat(3);  // "HiHiHi"

// isBlank() - 检查是否为空或仅包含空白字符
boolean blank1 = "".isBlank();      // true
boolean blank2 = "  ".isBlank();    // true
boolean blank3 = " a ".isBlank();   // false

// strip(), stripLeading(), stripTrailing() - 去除空白
String stripped = "  hello  ".strip();  // "hello"

// lines() - 分割为行流
String multiline = "Line1\nLine2\nLine3";
multiline.lines().forEach(System.out::println);

// indent() - 缩进
String indented = "Hello".indent(2);  // "  Hello\n"

// transform() - 转换
String transformed = "hello".transform(s -> s.toUpperCase());  // "HELLO"

// formatted() (Java 15+)
String formatted = "Value: %d".formatted(100);  // "Value: 100"
