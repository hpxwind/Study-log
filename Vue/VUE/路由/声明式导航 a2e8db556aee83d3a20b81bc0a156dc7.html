声明式导航

声明式导航是指通过在模板中使用特定的组件或指令来声明导航链接的方式。在 Vue Router 中，主要通过 <router-link>组件来实现声明式导航。

  • 核心特点
    • 声明式：告诉程序"要做什么"，而不是"怎么做"
    • 组件化：使用专门的导航组件
    • 响应式：自动处理活动状态样式
    • 语义化：代码可读性强
  • 声明式导航 vs 编程式导航
    特性声明式导航编程式导航
    实现方式<router-link>组件this.$router.push()等方法
    使用场景模板中的固定导航链接事件触发、条件跳转
    代码位置模板（Template）脚本（Script）
    活动状态自动处理需要手动处理
    可读性高，语义化强相对较低
  • 基本语法
    <template>
  <div>
    <!-- 最基本的声明式导航 -->
    <router-link to="/home">首页</router-link>
    
    <!-- 相当于传统的 a 标签 -->
    <a href="/home">首页（传统方式）</a>
  </div>
</template>

    <router-link>会被编译成<a>标签

    <!-- 编译前 -->
<router-link to="/home">首页</router-link>

<!-- 编译后 -->
<a href="/home" class="router-link-active">首页</a>
  • <router-link>的多种用法
    1. 字符串路径
    <template>
  <nav>
    <router-link to="/">首页</router-link>
    <router-link to="/about">关于</router-link>
    <router-link to="/contact">联系</router-link>
  </nav>
</template>
    1. 对象配置（常用，推荐）
    <template>
  <nav>
    <!-- 使用命名路由 -->
    <router-link :to="{ name: 'Home' }">首页</router-link>
    
    <!-- 使用路径 -->
    <router-link :to="{ path: '/about' }">关于</router-link>
    
    <!-- 带查询参数 -->
    <router-link :to="{ path: '/search', query: { keyword: 'vue', page: 1 } }">
      搜索Vue
    </router-link>
    
    <!-- 带动态参数 -->
    <router-link :to="{ name: 'User', params: { id: 123 } }">
      用户123
    </router-link>
  </nav>
</template>
    1. 动态路径参数
    <template>
  <div>
    <!-- 动态用户ID -->
    <router-link :to="`/user/${userId}`">我的主页</router-link>
    
    <!-- 对象形式的动态参数 -->
    <router-link :to="{ name: 'Product', params: { id: productId } }">
      {{ productName }}
    </router-link>
  </div>
</template>

<script>
export default {
  data() {
    return {
      userId: 123,
      productId: 456,
      productName: 'Vue教程'
    }
  }
}
</script>
  • 重要属性
    1. to属性
    <template>
  <div>
    <!-- 字符串形式 -->
    <router-link to="/user/123/profile">用户资料</router-link>
    
    <!-- 对象形式（更灵活） -->
    <router-link :to="{
      name: 'UserProfile',
      params: { id: 123 },
      query: { tab: 'settings' }
    }">用户设置</router-link>
  </div>
</template>
    1. 活动状态类名

    自动为当前活动路由添加样式类。

    <template>
  <nav>
    <router-link 
      to="/home"
      active-class="active-link"
      exact-active-class="exact-active-link"
    >
      首页
    </router-link>
  </nav>
</template>

    1. 精确匹配模式
    <template>
  <div>
    <!-- 精确匹配：只有路径完全相同时才激活 -->
    <router-link to="/" exact>首页（精确匹配）</router-link>
    
    <!-- 非精确匹配：路径包含时即激活 -->
    <router-link to="/user" active-class="active">用户（包含匹配）</router-link>
  </div>
</template>
    1. 自定义标签和样式
    <template>
  <div>
    <!-- 渲染为 button 标签 -->
    <router-link to="/home" tag="button" class="btn">
      按钮式导航
    </router-link>
    
    <!-- 渲染为 li 标签 -->
    <router-link to="/about" tag="li" active-class="active">
      <a>关于我们</a>
    </router-link>
    
    <!-- 自定义事件 -->
    <router-link to="/contact" event="dblclick">
      双击跳转
    </router-link>
  </div>
</template>

    作用域插槽

    <template>
  <div>
    <router-link 
      to="/dashboard" 
      custom
      v-slot="{ href, route, navigate, isActive, isExactActive }"
    >
      <a 
        :href="href" 
        @click="navigate"
        :class="{ 
          'nav-item': true, 
          'active': isActive,
          'exact-active': isExactActive 
        }"
      >
        📊 仪表板
        <span v-if="isActive" class="indicator">●</span>
      </a>
    </router-link>
  </div>
</template>

<script>
export default {
  methods: {
    handleNavigation(event, navigate) {
      // 自定义导航逻辑
      if (this.hasUnsavedChanges) {
        event.preventDefault()
        if (confirm('确定要离开吗？')) {
          navigate()
        }
      } else {
        navigate()
      }
    }
  }
}
</script>

    导航首位集成

    <template>
  <div>
    <router-link 
      to="/protected"
      @click.native="checkPermission"
    >
      需要权限的页面
    </router-link>
  </div>
</template>

<script>
export default {
  methods: {
    checkPermission(event) {
      if (!this.$store.state.isLoggedIn) {
        event.preventDefault()
        this.$router.push('/login')
      }
    }
  }
}
</script>

    2、面包屑导航

    <template>
  <nav class="breadcrumb">
    <router-link 
      v-for="(item, index) in breadcrumbs" 
      :key="index"
      :to="item.to"
      :class="['breadcrumb-item', { 'active': item.active }]"
    >
      {{ item.text }}
      <span v-if="!item.active" class="separator">/</span>
    </router-link>
  </nav>
</template>

<script>
export default {
  computed: {
    breadcrumbs() {
      const route = this.$route
      const crumbs = []
      
      // 首页面包屑
      crumbs.push({
        text: '首页',
        to: '/',
        active: route.path === '/'
      })
      
      // 动态生成面包屑
      if (route.path.includes('/product')) {
        crumbs.push({
          text: '产品详情',
          to: route.path,
          active: true
        })
      }
      
      return crumbs
    }
  }
}
</script>

    1. 性能优化

    <router-link 
      v-for="item in menuItems" 
      :key="item.id"
      :to="item.path"
      custom
      v-slot="{ isActive, navigate }"

      <button
