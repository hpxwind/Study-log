    // 包声明：定义这个类所在的文件夹路径，对应项目的目录结构
package com.hpxwind.tutorial.item;

// 导入模组主类，用于获取固定的MOD_ID（模组唯一标识）
import com.hpxwind.tutorial.TutorialMod;
// 导入Minecraft原版物品的基类，所有自定义物品都要继承这个类
import net.minecraft.world.item.Item;
// 导入Forge事件总线，用于注册物品到游戏
import net.minecraftforge.eventbus.api.IEventBus;
// 导入Forge的延迟注册器（推荐的注册方式，安全高效）
import net.minecraftforge.registries.DeferredRegister;
// 导入Forge的物品注册表（所有物品都在这里注册）
import net.minecraftforge.registries.ForgeRegistries;
// 导入注册对象，用于接收注册后的物品
import net.minecraftforge.registries.RegistryObject;

/**
 * 模组物品注册类
 * 作用：统一管理、注册所有【自定义物品】（如冰以太、原材料等）
 * 这是Forge模组开发的标准写法
 */
public class ModItems {
    /**
     * 创建物品【延迟注册器】
     * DeferredRegister：Forge推荐的注册工具，避免手动注册导致的重复/报错
     * 参数1：ForgeRegistries.ITEMS → 固定写法，代表注册【物品】
     * 参数2：TutorialMod.MOD_ID → 绑定你的模组唯一ID，防止和其他模组冲突
     */
    public static final DeferredRegister<Item> ITEMS =
            DeferredRegister.create(ForgeRegistries.ITEMS, TutorialMod.MOD_ID);

    /**
     * 注册第一个自定义物品：冰以太锭
     * ITEMS.register(注册名, 物品实例)
     * 注册名规则：只能用 小写字母、数字、下划线，**不能有大写/空格/斜杠/中文**
     * new Item.Properties() → 物品基础属性（默认属性，无特殊效果）
     */
    public static final RegistryObject<Item> ICE_ETHER =
            ITEMS.register("ice_ether", () -> new Item(new Item.Properties()));

    /**
     * 注册第二个自定义物品：原矿冰以太
     * 注册名 raw_ice_ether → 唯一不重复，和上面的 ice_ether 区分开
     */
    public static final RegistryObject<Item> RAW_ICE_ETHER =
            ITEMS.register("raw_ice_ether", () -> new Item(new Item.Properties()));

    // ==============================================
    // ❌ 【严重错误】注册名不能带斜杠 /！！！
    // 正确写法：cardboard 或 material_cardboard
    // ==============================================
    public static final RegistryObject<Item> CARDBOARD =
            ITEMS.register("material_cardboard", () -> new Item(new Item.Properties()));

    /**
     * 注册方法：将所有物品【提交到Forge事件总线】
     * 必须在模组主类中调用这个方法，物品才能生效
     * @param eventBus  Forge事件总线（从主类传入）
     */
    public static void register(IEventBus eventBus) {
        ITEMS.register(eventBus);
    }
}
register 名称参数只能 [a-z0-9/._-]

需要在物品栏中添加该物品，在TutorialMod.java中修改addCreative函数即可

  • TutorialMod.java
    // 方法作用：向【创造模式物品栏】中添加我们的自定义物品
// BuildCreativeModeTabContentsEvent：Forge 专门用于「修改创造模式标签页内容」的事件
private void addCreative(BuildCreativeModeTabContentsEvent event)
{
    // 判断：当前正在填充的创造模式标签页，是不是【原材料(INGREDIENTS)】标签
    // CreativeModeTabs.INGREDIENTS = 游戏默认的「材料」分类（和钻石、铁锭在一起）
    if (event.getTabKey() == CreativeModeTabs.INGREDIENTS) {
        
        // 把 冰以太 物品添加到「材料」标签页中
        event.accept(ModItems.ICE_ETHER);
        // 把 原矿冰以太 物品添加到「材料」标签页中
        event.accept(ModItems.RAW_ICE_ETHER);
        // 把 纸板 物品添加到「材料」标签页中
        event.accept(ModItems.CARDBOARD);
    }
}

创建延迟注册器（核心）

public static final DeferredRegister<Item> ITEMS =
        DeferredRegister.create(ForgeRegistries.ITEMS, TutorialMod.MOD_ID);
  1. DeferredRegister<Item>:Forge 官方延迟注册工具, Item 限定这个注册器只注册物品
  1. DeferredRegister.create:创建一个新的注册器，两个参数固定:
    1. ForgeRegistries.ITEMS:固定值，告诉游戏：我要注册的是物品
    1. TutorialMod.MOD_ID:模组命名空间[MOD_ID]=tutorial_mod(TutorialMod.java 中查看)

添加物品（注册单个物品）

public static final RegistryObject<Item> ICE_ETHER =
        ITEMS.register("ice_ether", () -> new Item(new Item.Properties()));
  1. RegistryObject<Item>:Forge 提供的物品包装类,安全获取注册后的物品，防止提前调用报错。
  1. ITEMS.register:往「物品清单」里添加一个物品，两个参数缺:
    1. "ice_ether”:物品注册名 （唯一且[a-z0-9/._-]）
    1. () -> new Item(new Item.Properties()):延迟创建物品对象，等到游戏注册时才生成，避免提前加载崩溃。
      1. new Item.Properties():物品基础属性（最大堆叠数、是否可合成等，默认 64 个）
      1. new Item(...):创建一个原版普通物品

提交注册（启动按钮）

public static void register(IEventBus eventBus) {
    ITEMS.register(eventBus);
}
  1. register(IEventBus eventBus):IEventBus=Forge 事件总线（游戏的中央调度器）
  1. ITEMS.register(eventBus):把我装满物品的「清单」提交给游戏

追踪register到 DeferredRegister.java/RegistryObject（Forge 底层注册物品的核心代码）中

其中ResourceLocation 函数生成模组物品的唯一身份标识：模组id+物品名（最容易出错），确保id和名字满足 [a-z0-9/._-]
// 1. 定义注册方法：传入物品名字、物品创建方式，返回包装好的物品
public <I extends T> RegistryObject<I> register(final String name, final Supplier<? extends I> sup)
{
    // 2. 检查：游戏已经开始注册了 → 不准再加新物品，直接报错
    if (seenRegisterEvent)
        throw new IllegalStateException("注册结束后不能新增物品！");

    // 3. 检查：物品名/创建方式不能为空，空了直接报错
    Objects.requireNonNull(name);
    Objects.requireNonNull(sup);

    // 4. 生成【唯一身份证】：模组ID + 物品名（比如 tutorial_mod:ice_ether）
    final ResourceLocation key = new ResourceLocation(modid, name);

    // 5. 创建物品的包装对象（方便游戏安全调用）
    RegistryObject<I> ret = RegistryObject.create(唯一身份证);

    // 6. ✨ 核心：把物品存进清单，如果名字重复 → 直接抛你之前的报错！
    if (entries.putIfAbsent(物品, 创建方式) != null) {
        throw new IllegalArgumentException("Duplicate registration " + name);
    }

    // 7. 返回注册好的物品
    return ret;
}

这是 Minecraft 中用于标识资源的核心类，其实现和用法在不同版本中有一些关键变化。但 implements Comparable<ResourceLocation> 这个特性几乎所有版本都保持一致。
