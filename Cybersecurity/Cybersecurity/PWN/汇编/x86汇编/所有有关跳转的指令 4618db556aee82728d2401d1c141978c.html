所有有关跳转的指令

  • 1、无条件跳转指令

    JMP直接跳转

    jmp label       ; 跳转到标签处
jmp eax         ; 跳转到EAX寄存器中的地址
jmp [ebx]       ; 跳转到EBX指向的内存地址中的地址
jmp short label ; 短跳转（-128到+127字节）
jmp near label  ; 近跳转（同一代码段内）
jmp far label   ; 远跳转（不同代码段）

  • 2、条件跳转指令
    • 1.基于单个标志位的跳转

      零标志位ZF：

      jz  label       ; ZF=1时跳转 (为零/相等)
jnz label       ; ZF=0时跳转 (非零/不相等)

      进位标志位CF :

      jc  label       ; CF=1时跳转 (有进位/借位)
jnc label       ; CF=0时跳转 (无进位/借位)

      符号标志位 SF：

      js  label       ; SF=1时跳转 (结果为负)
jns label       ; SF=0时跳转 (结果为非负)

      溢出标志位 OF：

      jo  label       ; OF=1时跳转 (有溢出)
jno label       ; OF=0时跳转 (无溢出)

      奇偶标志位 PF：

      jp  label       ; PF=1时跳转 (偶校验)
jnp label       ; PF=0时跳转 (奇校验)
jpe label       ; PF=1时跳转 (偶校验)  
jpo label       ; PF=0时跳转 (奇校验)
    • 2.基于无符号数比较的跳转
      ja  label       ; 高于跳转 (CF=0且ZF=0)
jae label       ; 高于或等于跳转 (CF=0)
jb  label       ; 低于跳转 (CF=1)
jbe label       ; 低于或等于跳转 (CF=1或ZF=1)

jnbe label      ; 不低于且不等于 (ja的别名)
jnb label       ; 不低于 (jae的别名)
jna label       ; 不高于 (jbe的别名)
jnae label      ; 不高于且不等于 (jb的别名)
    • 3.基于有符号数比较的跳转
      jg  label       ; 大于跳转 (ZF=0且SF=OF)
jge label       ; 大于或等于跳转 (SF=OF)
jl  label       ; 小于跳转 (SF≠OF)
jle label       ; 小于或等于跳转 (ZF=1或SF≠OF)

jnle label      ; 不小于且不等于 (jg的别名)
jnl label       ; 不小于 (jge的别名)
jng label       ; 不大于 (jle的别名)
jnge label      ; 不大于且不等于 (jl的别名)
    • 4.特殊条件跳转
      jcxz label      ; CX=0时跳转
jecxz label     ; ECX=0时跳转
jrcxz label     ; RCX=0时跳转 (x64)

loop label      ; CX/ECX/RCX减1，不为0则跳转
loope label     ; CX/ECX/RCX减1，ZF=1且不为0则跳转
loopz label     ; 同loope
loopne label    ; CX/ECX/RCX减1，ZF=0且不为0则跳转
loopnz label    ; 同loopne
  • 3、跳转指令详细
    • 1.比较和跳转
      ; 无符号数比较
mov eax, 10
mov ebx, 20
cmp eax, ebx    ; 比较EAX和EBX
jb  less        ; 如果EAX < EBX (无符号)则跳转
ja  greater     ; 如果EAX > EBX (无符号)则跳转
je  equal       ; 如果相等则跳转

; 有符号数比较
mov eax, -10
mov ebx, 20
cmp eax, ebx    ; 比较EAX和EBX
jl  less_signed ; 如果EAX < EBX (有符号)则跳转
jg  greater_signed ; 如果EAX > EBX (有符号)则跳转

; 测试零值
test eax, eax   ; 测试EAX是否为0
jz  zero        ; 如果EAX=0则跳转
jnz not_zero    ; 如果EAX≠0则跳转
    • 2.循环控制
      ; 使用LOOP指令的循环
mov ecx, 10     ; 循环10次
loop_start:
    ; 循环体代码
    loop loop_start ; ECX减1，如果≠0则跳转

; 使用条件跳转的循环
mov ecx, 10
loop_start2:
    ; 循环体代码
    dec ecx     ; ECX减1
    jnz loop_start2 ; 如果ECX≠0则继续循环
    • 3.复杂条件判断
      ; 检查范围: if (eax >= 0 && eax < 100)
cmp eax, 0
jl  out_of_range    ; 如果EAX < 0则跳转
cmp eax, 100
jge out_of_range   ; 如果EAX >= 100则跳转
; 在范围内的代码

; 多条件判断
cmp eax, ebx
je  equal_case      ; 相等的情况
ja  greater_case    ; EAX > EBX
jb  less_case       ; EAX < EBX
  • 4、跳转举例和编码
    • 1.跳转距离类型
      ; 短跳转: -128 到 +127 字节
jmp short label

; 近跳转: 同一代码段内，16位模式下±32KB，32位模式下±2GB
jmp near label

; 远跳转: 不同代码段
jmp far ptr label
jmp 1234h:5678h   ; 段:偏移
    • 2.相对跳转计算
      ; 编译器自动计算相对偏移
label1:
    mov eax, 1
    jmp label2     ; 相对跳转
    nop
    nop
label2:
    mov ebx, 2
  • 5、高级跳转技巧
    • 1.跳转表实现
      ; 使用跳转表实现switch-case
section .data
jump_table dd case0, case1, case2, case3

section .text
mov eax, [index]    ; 索引值
cmp eax, 3
ja  default_case    ; 如果索引>3则跳转到默认情况
jmp [jump_table + eax*4]  ; 跳转到对应case

case0:
    ; case 0 代码
    jmp end_switch
case1:
    ; case 1 代码  
    jmp end_switch
; ... 其他case
default_case:
    ; 默认情况代码
end_switch:
    • 2.条件设置指令
      ; SETcc指令 - 根据条件设置字节
cmp eax, ebx
setg cl        ; 如果EAX>EBX则CL=1，否则CL=0
sete dl        ; 如果相等则DL=1

; 用于实现布尔值
xor eax, eax
cmp ebx, ecx
sete al        ; 如果EBX=ECX则EAX=1，否则EAX=0
  • 6、常见错误和注意事项

    1.跳转范围错误

    ; 错误: 跳转距离超出短跳转范围
jmp short too_far_label  ; 可能汇编错误
; 正确: 让汇编器自动选择跳转类型
jmp too_far_label

    2.标志位污染

    ; 错误: 在条件跳转前修改了标志位
cmp eax, ebx
add ecx, 1      ; 修改了标志位！
jg greater      ; 现在判断的是ADD的结果，不是CMP的结果

; 正确: 保护标志位
cmp eax, ebx
pushf          ; 保存标志位
add ecx, 1
popf           ; 恢复标志位
jg greater

在64位模式下，跳转指令基本保持不变，但增加了：

jrcxz label ; RCX=0时跳转

总结表：

JE/JZZF=1相等/为零==
JNE/JNZZF=0不相等/非零≠≠
JA/JNBECF=0且ZF=0高于>-
JAE/JNBCF=0高于或等于≥-
JB/JNAECF=1低于<-
JBE/JNACF=1或ZF=1低于或等于≤-
JG/JNLEZF=0且SF=OF大于->
JGE/JNLSF=OF大于或等于-≥
JL/JNGESF≠OF小于-<
JLE/JNGZF=1或SF≠OF小于或等于-≤
