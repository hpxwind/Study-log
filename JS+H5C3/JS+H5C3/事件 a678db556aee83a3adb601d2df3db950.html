  • dom事件类型

    dom0类型-- 后面会覆盖前面

    dom2类型-- 绑定多个事件处理函数 按照顺序执行

    <div id="box"></div>
<div id="box2"></div>
<script>
    // dom0类型-- 后面会覆盖前面
    box.onclick = function() {
        console.log("1111")
    }

    // dom2类型-- 绑定多个事件处理函数 按照顺序执行
    box2.addEventListener("click", function() {
        console.log("222222")
    }) 

    box2.addEventListener("click", function() {
        console.log("333333")
    }) 
</script>
  • 事件类型
    • 1、鼠标事件
      事件类型触发条件特性说明
      click单击元素时触发最常用的事件
      dblclick双击元素时触发两次快速点击
      mousedown按下鼠标按钮时触发按下即触发
      mouseup释放鼠标按钮时触发在 mousedown 后触发
      mousemove鼠标在元素内移动时触发持续触发，注意性能优化
      mouseover鼠标移入元素或其子元素时触发会冒泡
      mouseout鼠标移出元素或其子元素时触发会冒泡
      mouseenter鼠标移入元素时触发不冒泡，不被子元素影响
      mouseleave鼠标移出元素时触发不冒泡，不被子元素影响
      contextmenu右键点击时触发可阻止默认右键菜单
      element.addEventListener('click', (e) => {
  console.log(`点击位置: (${e.clientX}, ${e.clientY})`);
});
    • 2、键盘事件
      事件类型触发条件特性说明
      keydown按下任意键时触发按住不放会持续触发
      keyup释放按键时触发在 keydown 后触发
      keypress按下字符键时触发（已废弃）不推荐使用，用 keydown 替代
      document.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') {
    console.log('回车键被按下');
  }
});
    • 3、表单事件
      事件类型触发条件特性说明
      submit表单提交时触发可调用 e.preventDefault() 阻止提交
      reset表单重置时触发
      change表单元素值改变并失焦后触发适用于输入框、下拉框等
      input输入框值变化时实时触发即时响应输入
      focus元素获得焦点时触发不冒泡
      blur元素失去焦点时触发不冒泡
      focusin元素即将获得焦点时触发会冒泡
      focusout元素即将失去焦点时触发会冒泡
      form.addEventListener('submit', (e) => {
  e.preventDefault(); // 阻止表单默认提交
  console.log('表单已提交');
});
    • 4、窗口事件
      事件类型触发条件
      resize浏览器窗口大小改变时触发
      scroll页面滚动时触发
      load页面及所有资源加载完成时触发
      DOMContentLoadedHTML 解析完成时触发（无需等待图片）
      beforeunload页面关闭前触发（可提示用户保存）
      hashchangeURL 的哈希值变化时触发
      window.addEventListener('resize', () => {
  console.log(`窗口大小: ${window.innerWidth}x${window.innerHeight}`);
});
    • 5、触摸事件
      事件类型触发条件
      touchstart手指触摸屏幕时触发
      touchmove手指在屏幕上滑动时触发
      touchend手指离开屏幕时触发
      touchcancel触摸被意外中断时触发（如来电）
      element.addEventListener('touchmove', (e) => {
  e.preventDefault(); // 阻止默认滚动行为
  console.log(`手指位置: (${e.touches[0].clientX}, ${e.touches[0].clientY})`);
});
    • 6、其他事件
      1. 媒体事件

        play, pause, ended（视频/音频播放控制）

      1. 剪贴板事件

        copy, cut, paste

      1. 拖拽事件

        dragstart, dragend, drop（需设置 draggable="true"）

      1. 动画事件

        animationstart, animationend

      1. 过渡事件

        transitionend

    • 7、事件委托

      通过冒泡机制将子元素事件委托给父元素处理:

      document.getElementById('parent').addEventListener('click', (e) => {
  if (e.target.matches('button.child')) {
    console.log('子按钮被点击');
  }
});
    • 8、事件对象常用属性
      属性/方法说明
      e.target触发事件的原始元素
      e.currentTarget当前处理事件的元素
      e.preventDefault()阻止默认行为（如链接跳转）
      e.stopPropagation()停止事件冒泡
      e.key键盘事件的按键值
      e.clientX / e.clientY鼠标/触摸点的视口坐标
  • 事件的传播(DOM事件流)（冒泡）

    当元素触发事件的时候，其父元素也会触发相同的事件，父元素的父元素也会触发相同的事件。

    如上图：

    如果点击了inner盒子上的点击事件，

    也就会触发outer上的点击事件，

    也会触发body上的点击事件，

    也会触发document上的点击事件,

    . . .

    • 标准事件流：先捕获到目标，再冒泡

      捕获：window → document → body → ..

      目标：inner

      冒泡：outer → .. → body → document → window

      默认：事件会在冒泡过程中触发

      如果想在捕获过程种看到，需要在dom2事件中绑定，并进行配置。

      <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        #outer {
            width:500px;
            height:500px;
            background-color:red;
            overflow: hidden;
        }

        #center {
            width:300px;
            height:300px;
            background-color:blue;
            overflow: hidden;
            margin: 20px;
        }

        #inner {
            width:100px;
            height:100px;
            background-color:yellow;
            overflow: hidden;
            margin: 40px;
        }
    </style>
</head>
<body>
    <div id="outer">
        <div id="center">
            <div id="inner"></div>
        </div>
    </div>

    <script>
        inner.onclick = function() { // 触发三个 inner center outer
            console.log("inner");
        }
        center.onclick = function() { // 触发两个 center outer
            console.log("center");
        }
        outer.onclick = function() { // 触发一个 outer
            console.log("outer");
        }
				
				 inner.addEeventListener("click", function() { // 接收第三个参数为true时，在捕获阶段触发
            console.log("inner 捕获");
        }, true);
    </script>
</body>
</html>

  • 事件委托

    事件委托（Event Delegation）是一种利用事件冒泡机制的技术，它允许我们将单个事件监听器附加到父元素，而不是为每个子元素分别添加监听器。

    // 1. 在父元素上添加事件监听
parentElement.addEventListener('click', function(event) {
  // 2. 检查事件源（实际被点击的元素）
  if (event.target.matches('.child-element')) {
    // 3. 执行针对子元素的处理逻辑
    console.log('子元素被点击');
  }
});

  • 案例
    • 右键菜单案例
      <!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>极简自定义右键菜单</title>
    <style>
        /* 自定义右键菜单样式 - 极简版本 */
        .custom-menu {
            position: fixed;
            background: white;
            border-radius: 6px;
            box-shadow: 0 3px 10px rgba(0, 0, 0, 0.15);
            z-index: 1000;
            display: none;
            min-width: 160px;
            overflow: hidden;
            border: 1px solid #e0e0e0;
        }
    </style>
</head>
<body>
    
    <!-- 自定义右键菜单 -->
    <div class="custom-menu" id="contextMenu">
        <div class="menu-item" data-action="copy">复制</div>
        <div class="menu-item" data-action="paste">粘贴</div>
        <div class="menu-item" data-action="forward">转发</div>
        <div class="menu-item" data-action="delete">删除</div>
        <div class="menu-item" data-action="select">全选</div>
    </div>
    
    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const contextMenu = document.getElementById('contextMenu');
            
            // 显示自定义右键菜单
            function showContextMenu(e) {
                e.preventDefault();
                
                // 显示菜单
                contextMenu.style.display = 'block';
                
                // 定位菜单
                const x = e.clientX; // 鼠标点击位置，相对于浏览器可视窗口左上角的水平距离
                const y = e.clientY;
                const windowWidth = window.innerWidth; // 获取浏览器可视窗口的宽度(不包括滚动条，边框)
                const windowHeight = window.innerHeight;
                const menuWidth = contextMenu.offsetWidth; // 获取contextMenu元素的实际宽度
                const menuHeight = contextMenu.offsetHeight;
                
                // 确保菜单不会超出窗口边界
                const adjustedX = x + menuWidth > windowWidth ? windowWidth - menuWidth - 10 : x;
                const adjustedY = y + menuHeight > windowHeight ? windowHeight - menuHeight - 10 : y;
                
                contextMenu.style.left = `${adjustedX}px`;
                contextMenu.style.top = `${adjustedY}px`;
            }
            
            // 隐藏自定义右键菜单
            function hideContextMenu() {
                contextMenu.style.display = 'none';
            }
            
            // 处理菜单项点击
            function handleMenuItemClick(e) {
                const action = e.target.dataset.action;
                
                if (!action) return;
                
                // 在控制台输出操作信息
                console.log(`执行操作: ${action}`);
                
                // 实际应用中，这里可以添加具体功能实现
                hideContextMenu();
            }
            
            // 事件监听
            document.addEventListener('contextmenu', showContextMenu);
            document.addEventListener('click', hideContextMenu);
            contextMenu.addEventListener('click', handleMenuItemClick);
        });
    </script>
</body>
</html>
