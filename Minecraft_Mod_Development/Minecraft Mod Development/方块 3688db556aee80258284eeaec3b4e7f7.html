方块类编写

在com.hpxwind.tutorial下创建block包并创建ModBlocks.java方块类

  • ModBlocks.java
    // 声明该类所在的包路径
package com.hpxwind.tutorial.block;

// 导入需要的依赖类（模组主类、物品注册类、MC原版类、Forge注册类）
import com.hpxwind.tutorial.TutorialMod;
import com.hpxwind.tutorial.item.ModItems;
import net.minecraft.world.item.BlockItem;
import net.minecraft.world.item.Item;
import net.minecraft.world.level.block.Block;
import net.minecraft.world.level.block.Blocks;
import net.minecraft.world.level.block.state.BlockBehaviour;
import net.minecraftforge.eventbus.api.IEventBus;
import net.minecraftforge.registries.DeferredRegister;
import net.minecraftforge.registries.ForgeRegistries;
import net.minecraftforge.registries.RegistryObject;

import java.util.function.Supplier;

/**
 * 模组所有方块的注册类
 * 作用：集中管理、注册所有自定义方块 + 自动绑定对应的方块物品
 */
public class ModBlocks {
    /**
     * 方块延迟注册器
     * DeferredRegister<Block>：专门用来注册【方块】的工具
     * ForgeRegistries.BLOCKS：固定写法，表示注册类型是方块
     * TutorialMod.MOD_ID：绑定你的模组ID，所有注册必须带
     */
    public static final DeferredRegister<Block> BLOCKS =
            DeferredRegister.create(ForgeRegistries.BLOCKS, TutorialMod.MOD_ID);

    /**
     * 注册第一个方块：冰乙太块（ICE_ETHER_BLOCK）
     * 名称：ice_ether_block（唯一ID，全小写，下划线）
     * new Block(...)：创建一个普通基础方块
     * BlockBehaviour.Properties：方块属性（硬度、抗性、声音、亮度等）
     * strength(破坏硬度, 爆炸抗性)
     */
    public static final RegistryObject<Block> ICE_ETHER_BLOCK =
            registerBlock("ice_ether_block", () -> new Block(BlockBehaviour.Properties.of().strength(1.5F, 3.0F)));

    // 注册第二个方块：原始冰乙太块
    public static final RegistryObject<Block> RAW_ICE_ETHER_BLOCK =
            registerBlock("raw_ice_ether_block", () -> new Block(BlockBehaviour.Properties.of().strength(1.5F, 3.0F)));

    /**
     * 注册第三个方块：冰乙太矿石
     * copy(Blocks.STONE)：复制【原版石头】的所有属性（硬度、抗性、声音等）
     * 不用自己写strength，直接抄石头
     */
    public static final RegistryObject<Block> ICE_ETHER_ORE =
            registerBlock("ice_ether_ore", () -> new Block(BlockBehaviour.Properties.copy(Blocks.STONE)));

    /**
     * 【自动给方块绑定物品】工具方法
     * 每个方块必须对应一个物品，才能拿在手里、放在快捷栏
     * 作用：把方块注册成物品，交给 ModItems.ITEMS 去统一注册
     */
    private static <T extends Block> void registerBlockItems(String name, RegistryObject<T> block) {
        // BlockItem = 方块对应的物品
        ModItems.ITEMS.register(name, () -> new BlockItem(block.get(), new Item.Properties()));
    }

    /**
     * 【统一注册方块】工具方法（核心简化代码）
     * 1. 注册方块本身
     * 2. 自动调用 registerBlockItems 给方块绑定物品
     * 3. 返回注册好的方块对象
     */
    private static <T extends Block> RegistryObject<T> registerBlock(String name, Supplier<T> block) {
        RegistryObject<T> blocks = BLOCKS.register(name, block); // 注册方块
        registerBlockItems(name, blocks); // 自动绑定物品
        return blocks;
    }

    /**
     * 最终注册：把所有方块提交给Forge事件总线
     * 必须在模组主类里调用这个方法
     */
    public static void register(IEventBus eventBus) {
        BLOCKS.register(eventBus);
    }
}

注册一个普通方块

public static final RegistryObject<Block> ICE_ETHER_BLOCK =
            registerBlock("ice_ether_block", 
            () -> new Block(BlockBehaviour.Properties.of()
            .strength(1.5F, 3.0F)));
  1. registerBlock(...):自己写的工具方法,自动注册方块 + 自动生成对应的物品.
  1. () -> new Block(...):Lambda 表达式，懒加载,需要时才创建方块对象
  1. BlockBehaviour.Properties.of():创建方块属性,决定方块：硬度、抗性、声音、亮度、是否可破坏等
  1. .strength(1.5F, 3.0F):分别是挖掘硬度，爆炸抗性。
public static final RegistryObject<Block> ICE_ETHER_ORE =
            registerBlock("ice_ether_ore", 
            () -> new Block(BlockBehaviour.Properties.copy(Blocks.STONE)));
  1. .copy(Blocks.STONE):复制原版石头的所有属性,不用自己写 .strength()
BlockBehaviour.Properties.of() → 从零新建一个空白属性
BlockBehaviour.Properties.copy() → 复制一个现成方块的所有属性
模型、材质靠 ID 自动对应

核心工具方法 1：自动给方块生成物品

private static <T extends Block> void registerBlockItems(String name, RegistryObject<T> block) {
    ModItems.ITEMS.register(name, () -> new BlockItem(block.get(), new Item.Properties()));
}
  1. <T extends Block>:方块物品都要继承方块类
  1. ModItems.ITEMS.register(...):把方块对应的物品注册到物品注册表,让你能在背包里拿到这个方块
  1. new BlockItem(...):BlockItem=方块对应的物品,没有它，你无法拿在手里、无法放置

核心工具方法 2：统一注册方块

private static <T extends Block> RegistryObject<T> registerBlock(String name, Supplier<T> block) {
    RegistryObject<T> blocks = BLOCKS.register(name, block);  // 注册方块
    registerBlockItems(name, blocks);                        // 自动绑定物品
    return blocks;                                           // 返回方块
}
