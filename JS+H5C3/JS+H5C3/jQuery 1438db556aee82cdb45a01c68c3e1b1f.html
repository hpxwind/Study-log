    1. 基本选择器

    选择器描述示例
    *选取所有元素$("*") 选取页面中所有元素
    element选取指定标签名的元素$("p") 选取所有 <p> 元素
    #id选取指定 ID 的元素$("#box") 选取 ID 为 "box" 的元素
    .class选取指定类名的元素$(".active") 选取所有类为 "active" 的元素
    selector1, selector2组合选择器，选取多个选择器匹配的元素$("div, p") 选取所有 <div> 和 <p> 元素

    2. 层次选择器

    选择器描述示例
    ancestor descendant选取祖先元素下的所有后代元素（包括子、孙等）$("div p") 选取 <div> 内所有 <p>（无论嵌套层级）
    parent > child选取父元素的直接子元素（仅一代）$("ul > li") 选取 <ul> 的直接子元素 <li>
    prev + next选取紧邻在 prev 元素后的 next 元素（同级）$("h1 + p") 选取 <h1> 后面紧邻的第一个 <p>
    prev ~ siblings选取 prev 元素后的所有同级 siblings 元素$("h1 ~ p") 选取 <h1> 后面所有同级的 <p>

    选择器描述示例
    :first选取第一个元素$("p:first") 选取第一个 <p>
    :last选取最后一个元素$("p:last") 选取最后一个 <p>
    :not(selector)排除匹配 selector 的元素$("input:not(:checked)") 选取未选中的 input
    :even选取索引为偶数的元素（从 0 开始）$("tr:even") 选取表格中偶数行
    :odd选取索引为奇数的元素$("tr:odd") 选取表格中奇数行
    :eq(index)选取索引等于 index 的元素$("li:eq(2)") 选取第 3 个 <li>（索引 2）

    是一组在 AJAX 请求生命周期不同阶段触发的回调函数，允许开发者在请求发送前、响应处理中、请求完成后等关键节点插入自定义逻辑，实现对 AJAX 行为的扩展或干预。

    • 核心作用

      拦截与修改：在请求发送前修改请求参数、 headers 等；在响应返回后预处理数据。

      统一处理：全局统一处理所有 AJAX 请求的错误、添加通用请求头（如 Token）、加载状态提示等。

      扩展功能：基于钩子实现自定义数据格式解析、请求重试、权限验证等。

    • 分类与执行时机

      jQuery 的 AJAX 钩子按作用范围分为 局部钩子（单个请求生效）和 全局钩子（所有请求生效），按执行阶段可分为以下几类：

      钩子类型执行时机核心参数作用场景示例
      beforeSend请求发送前（XMLHttpRequest已创建）xhr（请求对象）、settings（配置）添加请求头、取消请求（返回false）
      dataFilter响应数据接收后，success执行前data（原始响应）、type（数据类型）、xhr预处理响应（如解密、格式转换）
      success响应成功（HTTP 2xx + 数据解析成功）data（处理后数据）、status、xhr处理业务数据
      error请求失败（网络错误、HTTP 4xx/5xx、解析失败）xhr、status（错误类型）、error统一错误提示（如 “登录过期”）
      complete请求完成（无论成功 / 失败）xhr、status关闭加载动画、清理资源
      jsonpCallbackJSONP 请求中，自定义回调函数名-控制 JSONP 回调逻辑

    // 所有请求默认添加Content-Type

      xhr.setRequestHeader("Content-Type", "application/json");

    // 发送错误日志到后端
    $.post("/api/log/error", {
      url: xhr.responseURL,
      status: xhr.status,
      error: error,
      time: new Date().toISOString()
    });

  // 全局完成回调：隐藏加载动画
  complete: function(xhr, status) {
    $("#loading").hide();
  }

    针对 嵌套结构，如对象里套对象、数组里套数组：

    类型核心逻辑适用场景风险点
    浅复制只复制「第一层」数据，嵌套的子对象 / 子数组仍为 引用传递（修改会影响原数据）扁平结构（无嵌套的对象 / 数组）嵌套结构修改时会 “污染” 原数据
    深复制递归复制所有层级数据，子对象 / 子数组完全独立（修改不影响原数据）嵌套结构（如 {a: {b: 1}}）性能略低（需遍历所有层级）

    jQuery可以通过$.extend() 灵活实现深复制和浅复制，核心是是否传入true作为第一个参数

    • 浅复制$.extend(target, source1, source2, ...)

      语法：将 source 对象的属性复制到 target 对象（多个 source 时，后面的覆盖前面的）。

      核心：只复制第一层，嵌套子对象仍为引用。

      返回值：修改后的 target 对象（若未传 target，则自动创建空对象）。

      语法：第一个参数传 true，开启 “递归复制”，所有层级的子对象 / 子数组都会独立拷贝。

      • 浅复制：
        /**
 * 递归实现浅复制（仅处理对象、数组，不处理特殊类型）
 * @param {any} data - 要复制的数据（基本类型/对象/数组）
 * @returns {any} 复制后的新数据
 */
function shallowCopyRecursive(data) {
  // 1. 处理基本类型：直接返回（基本类型赋值即独立）
  if (typeof data !== 'object' || data === null) {
    return data;
  }

  // 2. 处理引用类型：仅创建顶层新实例，嵌套引用复用原引用
  let newData;
  // 判断是数组还是对象
  if (Array.isArray(data)) {
    newData = []; // 创建新数组
    // 遍历顶层元素：基本类型直接push，引用类型直接push原引用（不递归）
    data.forEach(item => newData.push(item));
  } else {
    newData = {}; // 创建新对象
    // 遍历顶层属性：基本类型直接赋值，引用类型直接赋值原引用（不递归）
    for (const key in data) {
      if (data.hasOwnProperty(key)) { // 只复制自身属性（排除原型链属性）
        newData[key] = data[key];
      }
    }
  }

  return newData;
}

        /**
 * 递归实现深复制（处理对象、数组，暂不处理特殊类型）
 * @param {any} data - 要复制的数据（基本类型/对象/数组）
 * @returns {any} 完全独立的新数据
 */
function deepCopyRecursive(data) {
  // 1. 处理基本类型：直接返回（基本类型无需递归）
  if (typeof data !== 'object' || data === null) {
    return data;
  }

  // 2. 处理引用类型：递归遍历每一层，创建新实例
  let newData;
  // 判断是数组还是对象
  if (Array.isArray(data)) {
    newData = [];
    // 遍历数组元素：递归复制每个元素（无论层级多深）
    data.forEach(item => newData.push(deepCopyRecursive(item)));
  } else {
    newData = {};
    // 遍历对象属性：递归复制每个属性值（无论层级多深）
    for (const key in data) {
      if (data.hasOwnProperty(key)) {
        newData[key] = deepCopyRecursive(data[key]);
      }
    }
  }

  return newData;
}

        /**
 * 全功能递归深复制（支持普通类型+特殊类型：函数、正则、Date、DOM元素）
 * @param {any} data - 要复制的数据
 * @returns {any} 完全独立的新数据
 */
function deepCopyFull(data) {
  // 1. 处理基本类型：直接返回
  if (typeof data !== 'object' || data === null) {
    return data;
  }

  // 2. 处理特殊引用类型：针对性创建新实例
  let newData;

  // 2.1 处理 Date 对象
  if (data instanceof Date) {
    newData = new Date(data); // 用原日期创建新 Date 实例
    return newData;
  }

  // 2.2 处理 RegExp 正则
  if (data instanceof RegExp) {
    // 提取原正则的模式（source）和修饰符（flags）
    const source = data.source;
    const flags = data.flags || data.toString().match(/\/([gimuy]*)$/)[1]; // 兼容旧浏览器
    newData = new RegExp(source, flags); // 创建新正则实例
    return newData;
  }

  // 2.3 处理 Function 函数（仅复制函数体，不保留闭包）
  if (typeof data === 'function') {
    // 解析函数参数和函数体（支持箭头函数和普通函数）
    const fnStr = data.toString();
    const match = fnStr.match(/^([^(]+)?\(([^)]*)\)\s*=>?\s*([\s\S]*)$/);
    if (!match) return data; // 复杂函数（如生成器函数）暂不处理，返回原引用
    const [, , argsStr, bodyStr] = match;
    // 处理箭头函数的 return 省略（如 x=>x+1 补为 return x+1）
    const body = bodyStr.includes('return') ? bodyStr : `return ${bodyStr}`;
    // 用 new Function 创建新函数实例
    newData = new Function(argsStr.split(',').map(arg => arg.trim()), body);
    return newData;
  }

  // 2.4 处理 DOM 元素（用 cloneNode 深复制结构，用 jQuery 补事件复制）
  if (data instanceof HTMLElement) {
    // 原生 cloneNode(true)：复制元素+所有子元素（但不复制事件）
    newData = data.cloneNode(true);
    // 若需复制事件，可结合 jQuery（需确保 data 已被 jQuery 包装）
    if (window.jQuery && data.jquery) {
      newData = jQuery(data).clone(true)[0]; // jQuery clone(true) 复制事件
    }
    return newData;
  }

  // 3. 处理普通引用类型：数组和对象（递归遍历）
  if (Array.isArray(data)) {
    newData = [];
    data.forEach(item => newData.push(deepCopyFull(item)));
  } else {
    newData = {};
    for (const key in data) {
      if (data.hasOwnProperty(key)) {
        newData[key] = deepCopyFull(data[key]);
      }
    }
  }

  return newData;
}

        避免循环引用（ Circular Reference ）若原数据存在 “循环引用”（如a = { b: a }），递归会陷入无限循环，导致栈溢出。

        function deepCopyWithCircular(data, map = new WeakMap()) {
  if (typeof data !== 'object' || data === null) return data;
  // 若已复制过该对象，直接返回副本（避免循环引用）
  if (map.has(data)) return map.get(data);

  let newData;
  // ... 其他类型处理逻辑（Date、RegExp 等）...

  // 记录当前对象到哈希表
  map.set(data, newData);

  // ... 递归遍历数组/对象 ...
  return newData;
}

        /**
 * 用栈实现浅复制（仅处理对象和数组，不递归嵌套）
 * @param {any} data - 待复制的数据
 * @returns {any} 浅复制结果
 */
function shallowCopyWithStack(data) {
  // 基本类型直接返回
  if (typeof data !== 'object' || data === null) {
    return data;
  }

  // 初始化栈：存储 [原数据, 复制后的数据]
  const stack = [];
  // 创建顶层副本（数组/对象）
  const result = Array.isArray(data) ? [] : {};
  stack.push([data, result]);

  // 处理栈中元素（仅遍历顶层）
  while (stack.length > 0) {
    const [original, copy] = stack.pop(); // 栈：后进先出

    // 遍历原数据的顶层属性/元素
    if (Array.isArray(original)) {
      original.forEach((item, index) => {
        // 浅复制：直接赋值（引用类型复用原引用）
        copy[index] = item;
      });
    } else {
      for (const key in original) {
        if (original.hasOwnProperty(key)) {
          // 浅复制：直接赋值（引用类型复用原引用）
          copy[key] = original[key];
        }
      }
    }
  }

  return result;
}

        /**
 * 用栈实现深复制（处理对象、数组、Date、RegExp）
 * @param {any} data - 待复制的数据
 * @returns {any} 深复制结果
 */
function deepCopyWithStack(data) {
  // 基本类型直接返回
  if (typeof data !== 'object' || data === null) {
    return data;
  }

  // 处理特殊类型
  if (data instanceof Date) {
    return new Date(data);
  }
  if (data instanceof RegExp) {
    return new RegExp(data.source, data.flags);
  }

  // 初始化栈和结果（数组/对象）
  const stack = [];
  const result = Array.isArray(data) ? [] : {};
  stack.push([data, result]);

  // 用栈迭代处理所有层级
  while (stack.length > 0) {
    const [original, copy] = stack.pop(); // 栈：后进先出

    // 遍历原数据的属性/元素
    if (Array.isArray(original)) {
      original.forEach((item, index) => {
        if (typeof item === 'object' && item !== null) {
          // 嵌套引用类型：创建新副本并压入栈，等待后续处理
          const childCopy = Array.isArray(item) ? [] : {};
          copy[index] = childCopy;
          stack.push([item, childCopy]); // 新元素入栈，继续深复制
        } else {
          // 基本类型：直接赋值
          copy[index] = item;
        }
      });
    } else {
      for (const key in original) {
        if (original.hasOwnProperty(key)) {
          const value = original[key];
          if (typeof value === 'object' && value !== null) {
            // 嵌套引用类型：创建新副本并压入栈
            const childCopy = Array.isArray(value) ? [] : {};
            copy[key] = childCopy;
            stack.push([value, childCopy]); // 新元素入栈
          } else {
            // 基本类型：直接赋值
            copy[key] = value;
          }
        }
      }
    }
  }

  return result;
}

        /**
 * 用队列实现深复制（广度优先遍历）
 * @param {any} data - 待复制的数据
 * @returns {any} 深复制结果
 */
function deepCopyWithQueue(data) {
  if (typeof data !== 'object' || data === null) {
    return data;
  }

  // 处理特殊类型
  if (data instanceof Date) return new Date(data);
  if (data instanceof RegExp) return new RegExp(data.source, data.flags);

  // 初始化队列和结果
  const queue = [];
  const result = Array.isArray(data) ? [] : {};
  queue.push([data, result]);

  // 用队列迭代处理所有层级
  while (queue.length > 0) {
    const [original, copy] = queue.shift(); // 队列：先进先出

    // 遍历属性/元素
    if (Array.isArray(original)) {
      original.forEach((item, index) => {
        if (typeof item === 'object' && item !== null) {
          const childCopy = Array.isArray(item) ? [] : {};
          copy[index] = childCopy;
          queue.push([item, childCopy]); // 新元素入队，继续处理
        } else {
          copy[index] = item;
        }
      });
    } else {
      for (const key in original) {
        if (original.hasOwnProperty(key)) {
          const value = original[key];
          if (typeof value === 'object' && value !== null) {
            const childCopy = Array.isArray(value) ? [] : {};
            copy[key] = childCopy;
            queue.push([value, childCopy]); // 新元素入队
          } else {
            copy[key] = value;
          }
        }
      }
    }
  }

  return result;
}

// 测试队列深复制
const original = {
  level: 1,
  next: {
    level: 2,
    next: { level: 3 }
  }
};
const copy = deepCopyWithQueue(original);
copy.next.next.level = 100;
console.log(original.next.next.level); // 3（原数据不受影响）

    无论是栈还是队列实现，都需解决循环引用（如 a = { b: a }）导致的无限循环问题。解决方案是用 WeakMap 记录已复制的对象：

    特性栈（深度优先）队列（广度优先）
    处理顺序先深入嵌套层级，再返回处理外层先处理外层，再逐层深入嵌套
    内存占用深层嵌套时栈可能堆积较多元素每层元素均匀分布，内存占用更稳定
    适用场景数据嵌套层级深但每层元素少（如树形结构）数据层级浅但每层元素多（如列表套列表）
    循环引用处理需用哈希表记录已复制对象，避免重复处理同左（所有迭代法都需处理循环引用）

    浅复制说白了就是只复制一层，对于第一层内的嵌套结构（数组，对象）复制的是引用，而不是数据。

    • 局限性，替代与性能建议
      1. jQuery $.extend() 的局限性：
        • 不支持复制 函数、正则表达式、Date 对象、DOM 元素 等特殊类型（复制后可能异常）。
        • 若需复制特殊类型，需结合原生方法（如 new Date(originalDate) 复制日期）。
      1. 替代方案：原生深复制（ES6+）

      若项目中没有强依赖 jQuery，可使用原生structuredClone()支持更多类型，如 Date、RegExp）：

      const obj = { date: new Date(), reg: /abc/ };
const deepCopy = structuredClone(obj); // 原生深复制，无需 jQuery

      注意：structuredClone() 不支持函数、DOM 元素，适合普通数据。

      3. 性能建议：

      • 扁平结构优先用浅复制（$.extend({}, obj) 或 [...arr]），性能更高。
      • 嵌套结构才用深复制（$.extend(true, {}, obj)），避免不必要的递归。

    对于复杂结构（函数、正则、DOM 元素 等特殊类型）→浅复制复用引用，深复制创建独立实例

    • 扩展元素方法（$.fn.extend()）

      用于添加 DOM 元素相关方法（类似 $().css()、$().html()），需通过 jQuery 选择器获取元素后调用，操作匹配的 DOM 元素。

      $.extend($.fn, {}) 和 $.fn.extend({}) 完全等价。

      $.fn.extend({
  插件名: function(参数) {
    // 遍历所有匹配元素（支持链式调用）
    return this.each(function() {
      // 对每个元素的操作（this 指向当前 DOM 元素）
      const $this = $(this); // 转为 jQuery 对象
      // 功能实现
    });
  }
});

      color: 'red',
      backgroundColor: 'yellow',
      fontSize: 'inherit'

    • jQuery插件开发规范

      1. 避免全局变量污染

      使用立即执行函数（IIFE）包裹插件代码，将 jQuery 作为参数传入，避免外部变量干扰。

      // 安全的插件包装方式
(function($) { // $ 作为参数，内部可安全使用
  // 插件代码（如 $.fn.extend(...)）
})(jQuery); // 传入全局 jQuery

      2. 支持默认配置与用户配置合并

      通过 $.extend() 合并默认配置（defaults）和用户传入的配置，提高灵活性。

      $.fn.myPlugin = function(options) {
  const defaults = { size: 10, color: 'black' };
  const settings = $.extend({}, defaults, options); // 浅合并
  // 若有嵌套配置，用深合并：$.extend(true, {}, defaults, options)
};

      3. 支持链式调用

      插件方法需返回 this（即 jQuery 对象），确保可以继续调用其他 jQuery 方法。

      $.fn.myPlugin = function() {
  return this.each(function() { // 返回 this 支持链式调用
    // 处理逻辑
  });
};

      4. 避免覆盖已有插件

      开发前检查插件名是否已存在，避免覆盖已有方法。

      if (!$.fn.highlight) { // 检查插件是否已存在
  $.fn.highlight = function() {
    // 插件逻辑
  };
}

      5. 事件命名空间

      绑定事件时使用命名空间（如 click.highlight），避免移除事件时影响其他逻辑。

      $.fn.highlight = function() {
  return this.each(function() {
    const $this = $(this);
    // 绑定带命名空间的事件
    $this.on('click.highlight', function() {
      $this.css('color', 'red');
    });
  });
};

// 后续可安全移除插件绑定的事件
$('div').off('click.highlight');

      6. 暴露公共方法与配置

      允许用户通过特定方法修改插件配置或调用内部功能。

      $.fn.tooltip = function(options) {
  const methods = {
    init: function() { /* 初始化逻辑 */ },
    show: function() { /* 显示提示 */ },
    hide: function() { /* 隐藏提示 */ }
  };

  // 支持通过字符串调用方法（如 $().tooltip('show')）
  if (typeof options === 'string' && methods[options]) {
    return methods[options].apply(this, Array.prototype.slice.call(arguments, 1));
  }

  // 默认执行初始化
  return methods.init.apply(this, arguments);
};

// 使用公共方法
$('.tip').tooltip(); // 初始化
$('.tip').tooltip('show'); // 调用显示方法
