    package test;

import java.io.File;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.net.InetAddress;
import java.net.ServerSocket;
import java.net.Socket;

/**
 * 自定义 HTTP 服务器类（核心启动类）
 * 作用：启动服务端，监听端口，接收浏览器请求，并分发处理请求和响应
 *
 * Custom HTTP Server Class (Core Startup Class)
 * Function: Start the server, listen on a port, accept browser requests,
 *           and distribute processing of requests and responses
 */
public class MyHttpServer {
    // 服务器监听端口号，默认 8080
    // Server listening port number, default 8080
    private int port = 8080;

    /**
     * 定义网站静态资源根目录
     * 路径：项目根目录 \WebContent
     * System.getProperty("user.dir") = 获取项目当前工作目录
     * File.separator = 系统自适应文件分隔符（Windows \，Linux /）
     *
     * Define the root directory of static web resources
     * Path: Project Root Directory \WebContent
     */
    public static String WebContent = System.getProperty("user.dir") + File.separator + "WebContent";

    /**
     * 接收客户端请求的核心方法
     * Core method to receive client requests
     */
    public void receiving() {
        // 声明服务器套接字：用于监听客户端连接
        // Declare server socket: used to listen for client connections
        ServerSocket serverSocket = null;

        try {
            // 1. 创建 ServerSocket 实例
            // 参数1：port 监听端口
            // 参数2：最大等待连接数（backlog）
            // 参数3：绑定本机回环地址 127.0.0.1，只允许本机访问
            serverSocket = new ServerSocket(port, 1, InetAddress.getByName("127.0.0.1"));
            System.out.println("简易服务器启动成功，监听端口：" + port);
        } catch (IOException e) {
            // 启动服务器失败，抛出异常
            // Failed to start the server, throw exception
            throw new RuntimeException(e);
        }

        // 2. 死循环：持续接收客户端请求（模拟 Tomcat 持续运行）
        // Infinite loop: Continuously accept client requests (simulate Tomcat running continuously)
        while (true) {
            // 客户端套接字：与浏览器建立连接后的通信通道
            // Client socket: Communication channel after establishing connection with browser
            Socket socket = null;
            // 输入流：读取浏览器发送的请求数据
            // Input stream: Read request data sent by the browser
            InputStream inputStream = null;
            // 输出流：向浏览器返回响应数据
            // Output stream: Return response data to the browser
            OutputStream outputStream = null;

            try {
                // 3. 阻塞监听：等待浏览器发起连接（没有连接就一直等）
                // Blocking listening: Wait for the browser to initiate a connection (wait if no connection)
                socket = serverSocket.accept();
                System.out.println("接收到新的客户端连接");

                // 4. 获取与该客户端通信的输入、输出流
                // Get input and output streams for communication with this client
                inputStream = socket.getInputStream();
                outputStream = socket.getOutputStream();

                // 5. 封装请求对象：解析浏览器发送的 HTTP 请求
                // Encapsulate request object: Parse HTTP request sent by browser
                MyHttpRequest req = new MyHttpRequest(inputStream);
                req.parse(); // 执行解析（解析出 URL）

                // 6. 封装响应对象：根据请求结果，返回数据给浏览器
                // Encapsulate response object: Return data to browser based on request result
                MyHttpResopnse response = new MyHttpResopnse(outputStream);
                response.response(req); // 执行响应（返回页面/404）

                System.out.println("请求处理完成：" + req.getUrl());

            } catch (IOException e) {
                // 请求处理过程中出现异常
                // Exception occurred during request processing
                e.printStackTrace();
                throw new RuntimeException(e);
            }
        }
    }
}

    package test;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;

/**
 * 自定义 HTTP 请求解析类
 * 作用：读取浏览器发送的 HTTP 请求数据，从中解析出请求的 URL 地址
 *
 * Custom HTTP Request Parsing Class
 * Function: Read HTTP request data sent by the browser and parse the requested URL address from it
 */
public class MyHttpRequest {

    // 输入流：用于接收浏览器发送的 HTTP 请求数据
    // Input stream: Used to receive HTTP request data sent by the browser
    private InputStream inputStream;

    // 存储解析出来的请求 URL 地址
    // Store the parsed request URL address
    private String url;

    /**
     * 构造方法
     * @param inputStream 传入的网络输入流（来自浏览器的请求数据）
     *
     * Constructor
     * @param inputStream Incoming network input stream (request data from the browser)
     */
    public MyHttpRequest(InputStream inputStream) {
        this.inputStream = inputStream;
    }

    /**
     * 解析 HTTP 请求的核心方法
     * 1. 从输入流中读取字节数据
     * 2. 将字节数据转为字符串
     * 3. 调用方法解析出 URL
     *
     * Core method to parse HTTP request
     * 1. Read byte data from input stream
     * 2. Convert byte data to string
     * 3. Call method to parse out URL
     */
    public void parse() {
        // 字符串缓冲区：用于拼接读取到的请求数据
        // String buffer: Used to concatenate the read request data
        StringBuffer sb = new StringBuffer();
        int i;
        // 字节缓冲区：一次性读取 2KB 的请求数据
        // Byte buffer: Read 2KB of request data at one time
        byte[] buffer = new byte[1024*2];
        try {
            // 从输入流中读取数据到缓冲区，返回读取到的字节数
            // Read data from input stream into buffer, return the number of bytes read
            i = inputStream.read(buffer);
        } catch (IOException e) {
            // 读取数据异常，抛出运行时异常
            // Exception occurred while reading data, throw runtime exception
            throw new RuntimeException(e);
        }

        // 将字节数组转为字符串（遍历字节，强转为字符并拼接）
        // Convert byte array to string (traverse bytes, cast to char and concatenate)
        for (int i1 = 0; i1 < i; i1++) {
            sb.append((char)buffer[i1]);
        }

        // 调用 parseUrl 方法，从完整请求头中解析出 URL 并赋值给成员变量
        // Call parseUrl method to parse URL from complete request header and assign to member variable
        url = parseUrl(sb.toString());
    }

    /**
     * 从 HTTP 请求头字符串中解析 URL
     * HTTP 请求头格式示例：GET /index.html HTTP/1.1
     * 解析逻辑：截取两个空格中间的路径部分
     *
     * Parse URL from HTTP request header string
     * HTTP request header example: GET /index.html HTTP/1.1
     * Parsing logic: Intercept the path part between two spaces
     * @param sb 完整的 HTTP 请求字符串 / Complete HTTP request string
     * @return 解析后的 URL 地址 / Parsed URL address
     */
    public String parseUrl(String sb) {
        int index1, index2;
        // 找到第一个空格的位置（GET 后面的空格）
        // Find the position of the first space (the space after GET)
        index1 = sb.indexOf(" ");
        // 从第一个空格后开始，找到第二个空格的位置（HTTP/1.1 前面的空格）
        // Start after the first space, find the position of the second space (the space before HTTP/1.1)
        index2 = sb.indexOf(" ", index1+1);

        // 截取两个空格中间的子串，即为请求的 URL
        // Intercept the substring between two spaces, which is the requested URL
        return sb.substring(index1+1, index2);
    }

    /**
     * 获取解析完成的 URL
     * @return 请求 URL
     *
     * Get the parsed URL
     * @return Request URL
     */
    public String getUrl(){
        return url;
    }
}

    package test;

import java.io.*;

/**
 * 自定义 HTTP 响应类
 * 作用：根据请求信息，向浏览器返回对应的 HTML 文件或 404 错误信息
 *
 * Custom HTTP Response Class
 * Function: Return corresponding HTML file or 404 error message to the browser based on request information
 */
public class MyHttpResopnse {
    // 输出流：用于向客户端（浏览器）发送响应数据
    // Output stream: Used to send response data to the client (browser)
    private OutputStream out;

    /**
     * 构造方法
     * @param out 传入的网络输出流
     *
     * Constructor
     * @param out Incoming network output stream
     */
    public MyHttpResopnse(OutputStream out) {
        this.out = out;
    }

    /**
     * 处理响应的核心方法
     * 1. 获取请求的 URL
     * 2. 读取服务器本地对应的 HTML 文件
     * 3. 封装 HTTP 响应协议并发送给浏览器
     *
     * Core method for processing response
     * 1. Get the requested URL
     * 2. Read the corresponding HTML file locally on the server
     * 3. Encapsulate HTTP response protocol and send it to the browser
     */
    public void response(MyHttpRequest request) throws IOException {
        // 字节缓冲区：用于临时读取文件数据
        // Byte buffer: Used to temporarily read file data
        byte[] bytes = new byte[1024*2];
        // 文件输入流：读取本地 HTML 文件
        // File input stream: Read local HTML files
        FileInputStream fis = null;

        // 获取请求的 URL 路径
        // Get the requested URL path
        String fisPath = request.getUrl();

        // 如果访问根路径 /，默认访问 index.html
        // If accessing the root path /, default to index.html
        if (fisPath.equals("/")) {
            fisPath = "/index.html";
        }

        // 存储最终响应字符串
        // Store the final response string
        String result = null;

        // 根据 Web 根目录 + 请求路径 创建文件对象
        // Create file object based on Web root directory + request path
        File file = new File(MyHttpServer.WebContent, fisPath);

        // 根据文件大小创建对应长度的字节数组
        // Create a byte array of corresponding length according to file size
        byte[] fileBytes = new byte[(int)file.length()];

        // 判断文件是否存在
        // Check if the file exists
        if (file.exists()) {
            // 文件存在 → 返回 200 + 文件内容
            // File exists → return 200 + file content
            fis = new FileInputStream(file);
            fis.read(bytes);       // 读取文件到缓冲区
            fis.close();          // 关闭流
            result = new String(fileBytes);  // 将文件字节转为字符串
            result = createMsg("200", result); // 封装 HTTP 响应头
            out.write(result.getBytes());    // 发送给浏览器
        } else {
            // 文件不存在 → 返回 404 错误
            // File does not exist → return 404 error
            String error = "404 File Not Found";
            String msg = createMsg("404", error);
            out.write(msg.getBytes());
        }
    }

    /**
     * 封装标准 HTTP 响应报文
     * 格式：协议版本 状态码 \r\n 响应头 \r\n 响应体
     *
     * Encapsulate standard HTTP response message
     * Format: Protocol version Status code \r\n Response header \r\n Response body
     * @param code 状态码：200=成功 404=未找到
     * @param message 响应内容
     * @return 完整 HTTP 响应字符串
     */
    public String createMsg(String code, String message) {
        return "HTTP/1.1 " + code + "\r\n"
                + "Content-Length:" + message.length() + "\r\n"
                + "\r\n"
                + message;
    }
}

│   │       │   └── web.xml   # Web 应用部署描述符
│   │       └── index.jsp      # 首页 JSP
│   └── test/              # 测试代码目录
├── target/                # 编译后的输出目录
├── pom.xml                # Maven 项目核心配置文件
└── 其他辅助文件（.gitignore、mvnw 等）

    项目根目录 javaweb1

    • javaweb1：你的项目名称，对应 Maven 的 artifactId。
    • .idea/：IDEA 自动生成的项目配置文件，包含运行配置、模块设置、本地历史等，不要手动修改，也不用提交到 Git。
    • .mvn/：Maven Wrapper 的目录，里面有自动下载 Maven 的脚本，确保不同环境下 Maven 版本一致。
    • pom.xml：Maven 项目的核心配置文件，定义了项目依赖（如 Servlet、JSP 等）、打包方式、插件配置等。
    • mvnw / mvnw.cmd：Maven Wrapper 的启动脚本，Windows 用 mvnw.cmd，Linux/Mac 用 mvnw，不用本地装 Maven 也能构建项目。
    • .gitignore：配置 Git 忽略的文件 / 目录，比如 target/、.idea/ 等，避免提交无关文件到仓库。

    2. src/main/java/ 目录（Java 源代码）

    • src/main/java/com.hpc.javaweb1/MyServlet.java：
      • 这是你的 Servlet 类，继承 HttpServlet，用来处理浏览器请求、生成响应。
      • 它的作用，就相当于你手写服务器里 MyHttpRequest + MyHttpResponse 的合体，由 Tomcat 帮你处理 Socket、流操作这些底层细节。

    3. src/main/resources/ 目录（资源文件）

    • 用来存放项目需要的配置文件，比如：
      • 数据库配置 db.properties
      • 日志配置 log4j.xml / logback.xml
      • 其他 XML/JSON 配置文件
    • Maven 编译时，会把这里的文件自动复制到 target/classes/ 目录下。

    4. src/main/webapp/ 目录（Web 应用根目录）

    这是 Web 项目的核心静态资源目录，Tomcat 部署时，这个目录就是 Web 应用的根路径。

    • index.jsp：项目的首页，浏览器访问 http://localhost:8080/javaweb1/ 时默认打开的页面。
    • WEB-INF/：
      • 这是一个受保护的目录，浏览器不能直接访问里面的文件，只能通过服务器内部访问。
      • web.xml：Web 应用的部署描述符文件，作用是：
        • 配置 Servlet、Filter、Listener
        • 设置欢迎页面（比如默认打开 index.jsp）
        • 配置错误页面、安全约束等
      • 在 Servlet 3.0 以后，很多配置可以用注解（@WebServlet）代替 web.xml，但旧项目或复杂配置还是会用到它。

    5. src/test/ 目录（测试代码）

    • src/test/java/：单元测试代码目录，比如用 JUnit 测试你的 Servlet 逻辑。
    • src/test/resources/：测试用的配置文件，和 main/resources 类似，只在测试时生效。

    6. target/ 目录（编译输出）

    这是 Maven 构建项目后生成的目录，存放编译后的字节码、打包后的文件。

    • classes/：编译后的 .class 文件和 resources 里的配置文件。
    • javaweb1-1.0-SNAPSHOT/：
      • 这是 Tomcat 部署时的实际应用目录结构，里面包含 WEB-INF/classes/、WEB-INF/lib/、index.jsp 等。
    • 打包后会生成 javaweb1-1.0-SNAPSHOT.war 文件，直接放到 Tomcat 的 webapps/ 目录下就能部署运行。

doGet 和 doPost 控制请求类型，get请求进doGet方法，post请求进doPost方法
HttpServletRequest 表示请求
HttpServletResponse 表示响应
整体是面向对象的思路

HttpServletRequest

表示请求，可以:

对象方法作用说明
HttpServletRequestreq.getParameter(name)接收客户端参数获取表单 / URL 传参，根据 input 的 name 取值
req.setCharacterEncoding("UTF-8")设置请求编码解决 POST 请求中文乱码
req.setAttribute(key, value)向 request 域存数据一次请求内有效
req.getAttribute(key)从 request 域取数据配合转发使用

     * Handle GET requests

     * 处理POST请求
     * form method="post" 提交时会进入这个方法
     *
     * Handle POST requests
     * This method is entered when form method="post" is submitted

    @Override

        // ========== 1. 设置请求编码（解决中文乱码）==========
        // 告诉服务器用UTF-8解析前端传递的参数

        // 告诉浏览器返回的是UTF-8格式的文本/html
        // Tell the browser that the returned content is UTF-8 formatted text/html
        resp.setContentType("text/html;charset=UTF-8");

        // 控制台打印日志
        // Console print log
        System.out.println("========= 进入doPost =========");

        // ========== 3. 获取前端表单传递的参数 ==========
        // 根据input标签的name属性获取值
        // Get values according to the name attribute of the input tag

        String name = req.getParameter("name");

        // 获取name="id"的输入框内容
        // Get the content of the input box with name="id"

        // Print received data to console

资源之间跳转有两种方法：1.转发 2.重定向
两者区别：
1.转发操作request,重定向操作response
2.转发浏览器地址栏不变,重定向浏览器地址栏会变
3.转发前后是同一个request对象，重定向前后是两个request对象
session和request一样，都是javaweb提供的对象，session用来存储登录用户信息
如果通过request进行数据的传递必须使用转发。如果是重定向会造成数据丢失(见上3.)
如果使用重定向进行跳转，同时需要数据共享，此时使用 session存储数据即可
session作用域大于request，不同的使用场景:
1.业务数据查询，后台查出数据，转发到页面进行展示，使用request存储数据
2.登录用户，多个场景中使用，session存储数据

    package com.hpc.javaweb1;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpSession;
import java.io.IOException;

/**
 * Login Servlet: Handle user login request
 * 登录处理器：处理用户登录请求
 */
@WebServlet("/login")
public class LoginServlet extends HttpServlet {

    /**
     * Handle POST request (Form submission)
     * 处理POST请求（表单提交）
     */
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        // Set request encoding to UTF-8 (Solve Chinese messy code)
        // 设置请求编码为UTF-8（解决中文乱码）
        req.setCharacterEncoding("UTF-8");

        // Get parameters from client form
        // 获取客户端表单传递的参数
        String username = req.getParameter("name");
        String password = req.getParameter("pwd");

        // Verify username
        // 验证用户名
        if (!username.equals(DBUtils.USERNAME)) {
            // Set error message to request field
            // 往request域存入错误提示信息
            req.setAttribute("msg_1", ResponseEnum.USERNAME_ERROR.getMsg());

            // Forward back to login page
            // 请求转发回到登录页面
            req.getRequestDispatcher("/login.jsp").forward(req, resp);
            return; // End method, stop execution
                    // 结束方法，终止后续执行
        }

        // Verify password
        // 验证密码
        if (!password.equals(DBUtils.PASSWORD)) {
            // Set error message to request field
            // 往request域存入错误提示信息
            req.setAttribute("msg_2", ResponseEnum.PASSWORD_ERROR.getMsg());

            // Forward back to login page
            // 请求转发回到登录页面
            req.getRequestDispatcher("/login.jsp").forward(req, resp);
            return; // End method, stop execution
                    // 结束方法，终止后续执行
        }

        // ===================== Login Success =====================
        // ===================== 登录成功 =====================

        // Get Session object
        // 获取会话对象Session
        HttpSession session = req.getSession();

        // Store username in Session (Global valid during login)
        // 将用户名存入Session（登录期间全局有效）
        session.setAttribute("name", username);

        // Set success tip
        // 设置成功提示信息
        req.setAttribute("msg_0", ResponseEnum.SUCCESS_TIP);

        // Forward to homepage
        // 转发到首页
        req.getRequestDispatcher("/index.jsp").forward(req, resp);
    }

    /**
     * Handle GET request
     * 处理GET请求
     */
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        // Unify processing logic, let GET call POST
        // 统一处理逻辑，让GET请求调用POST方法
        doPost(req, resp);
    }
}

数据库

Web: 前端(用户交互)→后端(业务逻辑)→数据库(数据持久化)

数据库使用与概念

JDBC

Java Database Connectivity 是一个独立于特定数据库系统的，通用的SQL数据库存取和操作的公共接口JDBC不仅仅针对于一种数据库，针对多种数据库,不同数据库都可以使用JDBC来管理。（JDBC 就是 Java 程序和数据库之间的 “翻译官 + 接线板”）

Java → MySQL适用的接口 → MySQL

Java → Oracle合适的接口 → Oracle

JDBC分为两部分：

  • 面向应用的API：开发者调用，接口
  • 面向数据库的API：给开发商开发数据库驱动程序

JDBC API

DriverManager类：管理各种不同的JDBC驱动

Connection接口：连接MySQL数据库

Statement接口：执行SQL语句

ResultSet接口：接收结果集

JDBC 原理

1、加载数据库驱动，Java程序的数据库的桥梁

Java → insert → MySQL驱动 → MySQL的SQL语句

2、获取Connection，一次连接

3、创建Statement，执行SQL

4、ResultSet接收结果

常见错误

1、Unknown database：数据库名称写错了

2、Access denied for user 'root'@'localhost' (using password: YES)：用户名或密码错误

3、No suitable driver for jdbc:mysql://localhost:3306/mytest1：数据库驱动未导入

4、The server time zone value is unrecognized：MySQL 时区和 JDK 不匹配

5、Cannot determine value type from string：字段是字符串，你用 getInt () 取

6、Parameter index out of range：索引越界 / 问号占位符数量不对

7、NullPointerException：空指针异常，连接没获取成功就使用，ResultSet 为空还调用 next ()

JDBC简单操作

  • 连接与CRUD操作
    import java.io.IOException;
import java.sql.*;

public class Main {
    public static void main(String[] args) throws Exception{
        // 连接数据库
        // url，name，pwd
        String url = "jdbc:mysql://127.0.0.1:3306/mytest1";
        String user = "root";
        String passwd = "123.com.cn";
        Connection connection = DriverManager.getConnection(url, user, passwd);
//        System.out.println(connection);

        // 创建sql
        // 添加 删除 更改
        String sql = "insert into user(name, pwd) values(?, ?)";
        PreparedStatement preparedStatement = connection.prepareStatement(sql);
        preparedStatement.setString(1, "bbb"); // 替换传递sql语句中的?
        preparedStatement.setString(2, "333"); // 类似的还有setInt setDouble setLong setBoolean ...
        int i = preparedStatement.executeUpdate(); // CRUD 中，在JDBC中update指 增删改 // 返回int类型，表示插入数据条数
        System.out.println(i);

        // 查询
        String sql = "select * from user where id = ?";
        PreparedStatement preparedStatement = connection.prepareStatement(sql);
        preparedStatement.setInt(1, 1);
        ResultSet resultSet = preparedStatement.executeQuery(); // 返回值不同，用ResultSet接收
        boolean next = resultSet.next();
        if (next) {
            int id =  resultSet.getInt(2);
            String name = resultSet.getString(1);
            String pwd = resultSet.getString(3);
            System.out.println(id + name + pwd);
        }
    }
}

        cpds.setAcquireIncrement(5);
        Connection connection = cpds.getConnection();

        QueryRunner runner = new QueryRunner();
        String sql = "select * from user where id = ?";
        // 可以直接返回一个对象，简化之前的一个个解析出来 // 注意未 BeanHandler 为查询单条数据

        String sql1 = "select * from user";
        // BeanListHandler 为查询多条数据
        List<User> list = runner.query(connection, sql1, new BeanListHandler<>(User.class));
        for (User l: list) {
            System.out.println(l);
        }
    }
}

    package db;

import org.apache.commons.dbutils.ResultSetHandler;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

/**
 * 自定义的 JDBC 查询工具类
 * 作用：简化 JDBC 操作，封装 预处理SQL、设置参数、执行查询、结果转换 等流程
 * 对应 Apache DBUtils 中的 QueryRunner
 */
public class MyQueryRunner {

    /**
     * 执行查询操作（SELECT语句）
     *
     * @param connection 数据库连接对象
     * @param sql        要执行的SQL语句，可以包含 ? 占位符
     * @param rsh        结果集处理器（MyBeanHandler / MyBeanListHandler），
     *                   负责将 ResultSet 转为 Java对象 或 List集合
     * @param params     为SQL中的 ? 占位符传递的参数，可变参数，个数不限
     * @return 返回处理好的结果对象（JavaBean 或 List）
     * @param <T> 泛型，让方法可以返回任意类型的结果
     * @throws MyException 自定义异常，参数非法时抛出
     */
    public <T> T query(Connection connection, String sql, ResultSetHandler<T> rsh, Object... params) throws MyException {

        // ======================== 1. 参数合法性校验（防御性编程） ========================
        // 如果数据库连接为空，抛出异常
        if (connection == null) {
            throw new MyException("[-] connection 数据库连接不能为空");
        }
        // 如果SQL语句为空，抛出异常
        if (sql == null) {
            throw new MyException("[-] sql 语句不能为空");
        }
        // 如果结果集处理器为空，无法封装结果，抛出异常
        if (rsh == null) {
            throw new MyException("[-] resultSetHandler 结果集处理器不能为空");
        }

        // 定义一个Object类型变量，用来接收最终封装好的结果（JavaBean / List）
        Object result = null;

        try {
            // ======================== 2. 获取预编译语句对象 ========================
            // connection.prepareStatement(sql)：
            // 对SQL进行预编译，返回 PreparedStatement 对象
            // 优点：防止SQL注入，效率更高
            PreparedStatement preparedStatement = connection.prepareStatement(sql);

            // ======================== 3. 为SQL中的 ? 占位符赋值 ========================
            // 如果用户传入了参数（params不为null），才进行赋值操作
            if (params != null) {
                // 获取参数的长度
                int len = params.length;
                // 遍历参数数组，依次为 ? 赋值
                for (int i = 0; i < len; i++) {
                    // 判断当前参数的具体类型，调用对应的set方法
                    // 如果参数是 Integer 类型
                    if (params[i] instanceof Integer) {
                        // 数据库下标从 1 开始，所以是 i+1
                        preparedStatement.setInt(i + 1, (Integer) params[i]);
                    }
                    // 如果参数是 String 类型
                    if (params[i] instanceof String) {
                        preparedStatement.setString(i + 1, (String) params[i]);
                    }
                    // 注意：这里只支持了int和string，如需支持更多类型需继续添加判断
                }
            }

            // ======================== 4. 执行SQL查询，获取结果集 ========================
            // executeQuery()：专门执行查询语句，返回 ResultSet（数据库原始查询结果）
            ResultSet rs = preparedStatement.executeQuery();

            // ======================== 5. 核心：结果集转换 ========================
            // 调用传入的处理器（MyBeanHandler/MyBeanListHandler）
            // 将 原始的ResultSet结果集 转换为 我们需要的Java对象
            result = rsh.handle(rs);

        } catch (SQLException e) {
            // 捕获SQL执行过程中产生的异常，包装为运行时异常抛出
            throw new RuntimeException("SQL执行异常：", e);
        }

        // ======================== 6. 返回最终结果 ========================
        // 将Object类型强转为泛型T，返回给调用者
        return (T) result;
    }
}

    package db;

import org.apache.commons.dbutils.ResultSetHandler;
import java.lang.reflect.Field;
import java.sql.ResultSet;
import java.sql.ResultSetMetaData;
import java.sql.SQLException;

/**
 * 自定义结果集处理器
 * 功能：将数据库查询返回的 **一行结果集（ResultSet）**
 *      自动封装成 **一个 JavaBean 对象（如 User 对象）**
 *
 * 实现 DBUtils 提供的 ResultSetHandler<T> 接口
 * T 代表要封装成的对象类型（例如 User）
 */
public class MyBeanHandler<T> implements ResultSetHandler<T> {

    // 保存要封装的实体类的 字节码对象（例如 User.class）
    // 作用：通过反射创建对象 + 给属性赋值
    private final Class<? extends T> type;

    /**
     * 构造方法
     * 使用时必须传入要封装的类，例如：
     * new MyBeanHandler<>(User.class)
     */
    public MyBeanHandler(Class<? extends T> type) {
        this.type = type;
    }

    /**
     * 核心方法：由 DBUtils 自动调用
     * 作用：把 ResultSet 结果集 → 转换成 JavaBean 对象
     * @param resultSet 数据库查询返回的结果集
     * @return 封装好的 JavaBean 对象
     */
    @Override
    public T handle(ResultSet resultSet) throws SQLException {

        // ==========================================
        // 1. 移动结果集指针到第一行
        // 如果没有数据（resultSet.next() = false），直接返回 null
        // ==========================================
        if (!resultSet.next()) {
            return null;
        }

        try {
            // ==========================================
            // 2. 通过反射 创建一个空的 JavaBean 对象
            // 等价于：User bean = new User();
            // ==========================================
            T bean = type.newInstance();

            // ==========================================
            // 3. 获取结果集元数据
            // 包含：列名、列数、列类型等信息
            // ==========================================
            ResultSetMetaData metaData = resultSet.getMetaData();

            // 获取当前结果集一共有多少列
            int columnCount = metaData.getColumnCount();

            // ==========================================
            // 4. 遍历每一列，给 JavaBean 属性赋值
            // 数据库下标从 1 开始，所以 i 从 1 开始
            // ==========================================
            for (int i = 1; i <= columnCount; i++) {

                // 获取当前列的 列名（如 id、name、pwd）
                String columnName = metaData.getColumnName(i);

                // 获取当前列的值（如 1、"张三"、"123456"）
                Object value = resultSet.getObject(i);

                try {
                    // ======================================
                    // 反射：根据列名找到实体类里对应的属性
                    // 例如列名 id → 找到 User 类的 id 字段
                    // ======================================
                    Field field = type.getDeclaredField(columnName);

                    // 开启暴力反射：允许访问 private 修饰的属性
                    field.setAccessible(true);

                    // ======================================
                    // 给对象的属性赋值
                    // 等价于：user.setId(1) / user.setName("张三")
                    // ======================================
                    field.set(bean, value);

                } catch (NoSuchFieldException ignored) {
                    // 如果列名和属性名不匹配，直接忽略，不报错
                }
            }

            // 打印最终封装好的对象（方便调试）
            System.out.println(bean);

            // 返回封装完成的 JavaBean 对象
            return bean;

        } catch (Exception e) {
            // 捕获反射异常，统一抛出运行时异常
            throw new RuntimeException("Bean封装失败", e);
        }
    }
}

    package db;

import org.apache.commons.dbutils.ResultSetHandler;
import java.lang.reflect.Field;
import java.sql.ResultSet;
import java.sql.ResultSetMetaData;
import java.sql.SQLException;
import java.util.ArrayList;
import java.util.List;

/**
 * 自定义结果集处理器
 * 作用：将 JDBC 查询返回的 ResultSet（多行数据） 转换为 List<JavaBean>
 * 实现 Apache DBUtils 提供的 ResultSetHandler 接口
 * T：要封装的 JavaBean 类型（如 User、Admin 等）
 */
public class MyBeanListHandler<T> implements ResultSetHandler<List<T>> {

    // 用来存储要封装的 字节码对象，例如 User.class
    // final：一旦赋值不能修改
    private final Class<? extends T> type;

    /**
     * 构造方法
     * @param type 传入你要封装的实体类的字节码对象（User.class）
     */
    public MyBeanListHandler(Class<? extends T> type) {
        this.type = type;
    }

    /**
     * 核心重写方法：真正执行 结果集 → List<T> 的转换
     * @param rs 数据库查询后返回的原始结果集（包含多行数据）
     * @return 返回封装好的 JavaBean 集合 List<T>
     * @throws SQLException SQL 异常
     */
    @Override
    public List<T> handle(ResultSet rs) throws SQLException {
        // 1. 创建一个空集合，用来存放封装好的多个 JavaBean 对象
        List<T> list = new ArrayList<>();

        // 2. 【核心循环】遍历结果集的每一行
        // rs.next()：每次调用，光标向下移动一行
        // 有数据返回 true，继续循环；没有数据返回 false，结束循环
        while (rs.next()) {
            try {
                // 3. 通过【反射】创建一个空的 JavaBean 对象
                // 等价于：User bean = new User();
                T bean = type.newInstance();

                // 4. 获取结果集的【元数据】（包含：列名、列数、列类型等信息）
                ResultSetMetaData metaData = rs.getMetaData();

                // 5. 获取当前结果集总共有多少列
                int columnCount = metaData.getColumnCount();

                // 6. 【内层循环】遍历当前行的每一列，给 JavaBean 赋值
                // 数据库列下标从 1 开始，所以 i 从 1 开始
                for (int i = 1; i <= columnCount; i++) {
                    // 根据下标获取【数据库列名】，如 id、name、pwd
                    String columnName = metaData.getColumnName(i);

                    // 根据下标获取【当前列的值】
                    Object value = rs.getObject(i);

                    try {
                        // 7. 通过【反射】：根据列名找到实体类中对应的属性
                        Field field = type.getDeclaredField(columnName);

                        // 暴力反射：开启访问权限，允许给 private 修饰的属性赋值
                        field.setAccessible(true);

                        // 8. 【核心赋值】给 JavaBean 的属性设置值
                        // 等价于：user.setId(1) / user.setName("张三")
                        field.set(bean, value);

                    } catch (NoSuchFieldException ignored) {
                        // 如果数据库列名 和 实体类属性名不匹配，直接忽略，不报错
                    }
                }

                // 9. 把封装好的 1 个 JavaBean 对象，添加到集合中
                list.add(bean);

            } catch (Exception e) {
                // 捕获反射创建对象时的异常，包装后抛出
                throw new SQLException("Bean 封装失败", e);
            }
        }

        // 打印最终封装好的集合（方便调试）
        System.out.println("封装结果：" + list);

        // 10. 返回装满了多个 JavaBean 的 List 集合
        return list;
    }
