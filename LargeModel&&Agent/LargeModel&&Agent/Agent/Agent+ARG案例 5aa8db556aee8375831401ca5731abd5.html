需求与服务

  1. 智能问答服务
  • 处理购买前的产品咨询（如功能、价格、对比等）。
  • 解决购买后的使用问题（如操作指导、故障处理、维护建议等）。
  • 基于RAG技术，从知识库中检索准确信息并生成自然语言回答，确保响应及时且可靠。
  1. 使用报告与优化建议生成：
  • 针对已购买用户，自动分析扫地机器人的使用数据（如清洁频率、耗材状态、错误日志等）。
  • 生成个性化报告，总结使用情况并提供优化建议（如清洁计划调整、部件更换提醒等）。
  • 支持用户主动查询报告或系统定期推送，帮助用户最大化产品价值。

    import os
import hashlib
from logger_handler import logger
from langchain_core.documents import Document
from langchain_community.document_loaders import PyPDFLoader, TextLoader


def get_file_md5_hex(filepath: str):
    """
    Calculate the MD5 hash of a file for duplicate detection.
    :param filepath: Path to the target file
    """
    if not os.path.exists(filepath):
        logger.error(f"[-][get_file_md5_hex] File {filepath} does not exist")

    if not os.path.isfile(filepath):
        logger.error(f"[-][get_file_md5_hex] Path {filepath} is not a file")

    md5_o = hashlib.md5()
    chunk_size = 1024 * 4

    try:
        # Must read in binary mode
        with open(filepath, "rb") as f:
            while chunk := f.read(chunk_size):
                # Walrus operator: assign first, then evaluate
                md5_o.update(chunk)
                md5_hex = md5_o.hexdigest()
    except Exception as e:
        logger.error(f"[-][get_file_md5_hex] Failed to calculate MD5 for {filepath}: {e}")


def listdir_with_allowed_type(path: str, allowed_types: tuple[str]):
    """
    List files in a directory that match allowed file extensions.
    :param path: Target directory path
    :param allowed_types: Tuple of allowed file extensions (e.g., (".txt", ".pdf"))
    :return: Tuple of valid file paths
    """
    files = []
    if not os.path.isdir(path):
        logger.error(f"[-][listdir_with_allowed_type] Path {path} is not a directory")
        return allowed_types

    for f in os.listdir(path):
        if f.endswith(allowed_types):
            files.append(os.path.join(path, f))

    # Convert to tuple to prevent modification
    return tuple(files)


def pdf_loader(filepath: str, passw=None) -> list[Document]:
    """
    Load and parse a PDF file into Document objects.
    :param filepath: Path to the PDF file
    :param passw: Optional password for encrypted PDFs
    :return: List of LangChain Document objects
    """
    return PyPDFLoader(filepath, passw).load()


def txt_loader(filepath: str) -> list[Document]:
    """
    Load and parse a text file into Document objects.
    :param filepath: Path to the text file
    :return: List of LangChain Document objects
    """
    return TextLoader(filepath, encoding="utf-8").load()

import logging

# Root directory for storing log files
LOG_ROOT = get_abs_path("logs")

# Log format: [time] - [logger name] - [level] - [file:line] - [message]
DEFAULT_LOG_FORMAT = logging.Formatter(
    "%(asctime)s - %(name)s - %(levelname)s - %(filename)s:%(lineno)d - %(message)s"
)


def get_logger(
        name: str = "agent",
        console_level: int = logging.INFO,
        file_level: int = logging.DEBUG,
        log_file=None
) -> logging.Logger:
    """
    Create and configure a logger with both console and file output.
    :param name: Logger name
    :param console_level: Log level for console output
    :param file_level: Log level for file output
    :param log_file: Custom log file path (optional)
    :return: Configured logger instance
    """
    logger = logging.getLogger(name)
    logger.setLevel(logging.DEBUG)

    # Avoid adding duplicate handlers
    if logger.handlers:
        return logger

    # Console handler
    console_handler = logging.StreamHandler()
    console_handler.setLevel(console_level)
    console_handler.setFormatter(DEFAULT_LOG_FORMAT)
    logger.addHandler(console_handler)

    # File handler
    if not log_file:
        log_file = os.path.join(LOG_ROOT, f"{name}_{datetime.now().strftime('%Y%m%d')}.log")

    file_handler = logging.FileHandler(log_file, encoding="utf-8")
    file_handler.setLevel(file_level)
    file_handler.setFormatter(DEFAULT_LOG_FORMAT)
    logger.addHandler(file_handler)

    return logger


# Quick access to the default logger

if __name__ == "__main__":
    logger.info("Info log test")

rag向量存储(vector_store)

RAG 系统的向量库服务类，基于 Chroma 实现本地知识库管理。

它自动扫描指定目录下的 TXT、PDF 文件，通过 MD5 校验避免重复导入，读取文件内容后用文本分割器将长文档切分为适配大模型的小块，再通过嵌入模型向量化并持久化存储。

类中提供检索器接口，采用 MMR 算法返回最相关的 3 条文档，为上层问答服务提供资料支撑。

整体负责文档入库、切片、向量化、去重与检索，是 RAG 系统的核心数据支撑模块。

    import os
from langchain_chroma import Chroma
from 大模型.Agent案例.utils.config_handler import chroma_conf
from 大模型.Agent案例.model.factory import embed_model
from 大模型.Agent案例.utils.path_tool import get_abs_path
from 大模型.Agent案例.utils.logger_handler import logger
from 大模型.Agent案例.utils.file_handler import (
    pdf_loader,
    txt_loader,
    listdir_with_allowed_type,
    get_file_md5_hex
)
from langchain_text_splitters import RecursiveCharacterTextSplitter


class VectorStoreService:
    """
    Vector Database Service Class
    Main functions: load documents -> split text -> embedding -> store into Chroma -> similarity retrieval
    """

    def __init__(self):
        """
        Initialize vector database and text splitter
        Step 1: Create or connect Chroma vector database
        Step 2: Initialize text chunk splitter for long document segmentation
        """
        # Initialize Chroma vector store
        self.vector_store = Chroma(
            collection_name=chroma_conf["collection_name"],
            embedding_function=embed_model,
            persist_directory=chroma_conf["persist_directory"],
        )

        # Initialize recursive character text splitter
        self.spliter = RecursiveCharacterTextSplitter(
            chunk_size=chroma_conf["chunk_size"],
            chunk_overlap=chroma_conf["chunk_overlap"],
            separators=chroma_conf["separators"],
        )

    def get_retriever(self):
        """
        Get document retriever for similarity search
        search_type="mmr": Max Marginal Relevance, return more diverse results
        k: return top 3 related docs
        fetch_k: select candidates from top 10 docs
        """
        return self.vector_store.as_retriever(
            search_type="mmr",
            search_kwargs={"k": 3, "fetch_k": 10}
        )

    def load_document(self):
        """
        Core method: Batch load local files into vector database
        Workflow:
        1. Traverse allowed files in data folder
        2. Calculate file MD5 to avoid duplicate storage
        3. Load txt/pdf content
        4. Split long text into small chunks
        5. Add chunks to Chroma vector store
        6. Record MD5 of processed files
        """

        def check_md5_hex(md5_for_check: str):
            """
            Check if current file has been processed before via MD5 record
            :param md5_for_check: md5 string of current file
            :return: True -> already processed; False -> new file
            """
            md5_file_path = get_abs_path(chroma_conf["md5_hex_store"])
            # Create md5 record file if not exists
            if not os.path.exists(md5_file_path):
                open(md5_file_path, "w", encoding='utf-8').close()
                logger.error(f"[-][load_document->check_md5_hex]{chroma_conf['md5_hex_store']} not found, auto created")
                return False

            # Read historical md5 records line by line
            with open(md5_file_path, "r", encoding='utf-8') as f:
                for line in f.readlines():
                    line = line.strip()
                    if line == md5_for_check:
                        return True
                return False

        def save_md5_hex(md5_for_check: str):
            """
            Save md5 of processed file into record file
            Avoid repeated loading next time
            """
            md5_file_path = get_abs_path(chroma_conf["md5_hex_store"])
            with open(md5_file_path, "a", encoding='utf-8') as f:
                f.write(md5_for_check + "\n")

        def get_file_documents(read_path: str):
            """
            Load file content according to file suffix
            Support: .txt and .pdf
            :param read_path: absolute path of target file
            :return: LangChain Document list
            """
            if read_path.endswith("txt"):
                return txt_loader(read_path)
            if read_path.endswith("pdf"):
                return pdf_loader(read_path)
            return []

        # Get all allowed type files under data directory
        allowed_files_path: list[str] = listdir_with_allowed_type(
            get_abs_path(chroma_conf["data_path"]),
            tuple(chroma_conf["allow_knowledge_file_type"]),
        )

        # Traverse every qualified file
        for path in allowed_files_path:
            # Calculate md5 value of current file
            md5_hex = get_file_md5_hex(path)

            # Skip if md5 calculate failed
            if not md5_hex:
                logger.error(f"[-][load_document]{path} MD5 calculate failed, skip")
                continue

            # Skip if file already exists in vector store
            if check_md5_hex(md5_hex):
                logger.info(f"[+][load_document]{path} already in knowledge base, skip")
                continue

            try:
                # Load raw document content
                documents = get_file_documents(path)

                # Skip if no valid content
                if not documents:
                    logger.warning(f"[-][load_document]{documents} no valid text, skip")
                    continue

                # Split long document into small chunks
                split_document = self.spliter.split_documents(documents)

                # Skip if split result is empty
                if not split_document:
                    logger.warning(f"[-][load_document]{documents}-{path} split empty, skip")
                    continue

                # Add split chunks into Chroma vector database
                self.vector_store.add_documents(split_document)
                # Record md5 to mark file as processed
                save_md5_hex(md5_hex)
                logger.info(f"[+][load_document]{documents}-{path} loaded to knowledge base successfully")

            except Exception as e:
                logger.error(f"[-][load_document]{path} load failed: {str(e)}", exc_info=True)

rag本地知识库寻找服务(rag_service)

RAG 智能问答总结服务核心类，依托向量库实现知识库问答。

服务初始化时加载向量库检索器、提示词模板与大模型，构建完整推理链路。

用户提问后，自动从向量库检索相关参考资料，将资料与问题拼接为规范上下文，传入大模型生成回答。

内置调试函数打印完整提示词，便于开发排查。

该类是 RAG 系统的交互入口，负责整合检索与生成能力，让 AI 基于本地知识库准确、简洁地总结并回复用户问题。

    """
Summary Service Class: User asks a question, searches reference materials,
submits the question and materials to the LLM, and gets a summarized response.
总结服务类：用户提问，搜索参考资料，将提问和参考资料提交给模型，让模型总结回复
"""
from langchain_core.documents import Document
from langchain_core.output_parsers import StrOutputParser
from vector_store import VectorStoreService
from 大模型.Agent案例.utils.prompt_loader import load_rag_prompts
from langchain_core.prompts import PromptTemplate
from 大模型.Agent案例.model.factory import chat_model


def print_prompt(prompt):
    """
    Helper function to print the complete prompt content for debugging
    辅助函数：打印完整提示词内容，用于调试查看
    Args:
        prompt: Formatted prompt template object 格式化后的提示词模板对象
    Returns:
        Original prompt object 原始提示词对象
    """
    print("=" * 20)
    print(prompt.to_string())
    print("=" * 20)
    return prompt


class RagSummarizeService(object):
    """
    RAG-based summarization service class
    基于RAG的问答总结服务类
    Core functions: Retrieve relevant docs → Generate answer with LLM
    核心功能：检索相关文档 → 调用大模型生成答案
    """

    def __init__(self):
        """
        Class initialization method
        类初始化方法
        Initialize vector store, retriever, prompt, LLM and inference chain
        初始化向量库、检索器、提示词、大模型和推理链
        """
        # Initialize vector storage service 初始化向量存储服务
        self.vector_store = VectorStoreService()
        # Get document retriever from vector store 从向量库获取文档检索器
        self.retriever = self.vector_store.get_retriever()
        # Load RAG prompt template text 加载RAG提示词模板文本
        self.prompt_text = load_rag_prompts()
        # Convert text to LangChain PromptTemplate 转换为LangChain提示词模板对象
        self.prompt_template = PromptTemplate.from_template(self.prompt_text)
        # Bind large language model 绑定大语言模型
        self.model = chat_model
        # Build the complete inference chain 构建完整的推理调用链
        self.chain = self.__init__chain()

    def __init__chain(self):
        """
        Private method to build the LLM inference chain
        私有方法：构建大模型推理调用链
        Chain structure: Prompt → Print Debug → LLM → String Output Parser
        链结构：提示词 → 调试打印 → 大模型 → 字符串解析器
        Returns:
            Callable LangChain chain 可调用的LangChain链
        """
        chain = self.prompt_template | print_prompt | self.model | StrOutputParser()
        return chain

    def retriever_docs(self, query: str) -> list[Document]:
        """
        Retrieve relevant documents from vector store by user query
        根据用户问题从向量库检索相关参考文档
        Args:
            query: User's question string 用户的提问文本
        Returns:
            List of matched Document objects 匹配到的文档对象列表
        """
        return self.retriever.invoke(query)

    def rag_summarize(self, query: str) -> str:
        """
        Core method: Execute RAG process and return summarized answer
        核心方法：执行RAG流程并返回总结后的答案
        Step 1: Retrieve relevant docs 步骤1：检索相关文档
        Step 2: Concatenate docs into context string 步骤2：拼接文档为上下文文本
        Step 3: Invoke LLM chain and get response 步骤3：调用模型链获取结果
        Args:
            query: User input question 用户输入的问题
        Returns:
            Final summarized answer 最终总结生成的答案
        """
        # Step 1: Get relevant documents from vector database
        # 步骤1：从向量数据库获取相关参考文档
        context_docs = self.retriever_docs(query)

        # Step 2: Concatenate all documents into a single context string
        # 步骤2：将所有检索到的文档拼接成上下文字符串
        context = ""
        counter = 0
        for doc in context_docs:
            counter += 1
            # Add document content and metadata to context
            # 将文档内容和元数据加入上下文
            context += f"[参考资料:{counter}] 参考资料:{doc.page_content} | 参考元数据: {doc.metadata}\n"

        # Step 3: Pass query and context to LLM chain and get response
        # 步骤3：将问题和上下文传入模型链，获取最终回答
        return self.chain.invoke(
            {
                "input": query,
                "context": context,
            }
        )

def get_current_month() -> str:
    """
    Load external CSV data into global dictionary (one-time loading)
    将外部CSV数据加载到全局字典（仅加载一次）
    Steps:
        1. Get external data path  获取外部数据路径
        2. Check file existence    检查文件是否存在
        3. Read and parse CSV      读取并解析CSV
        4. Store in global dict    存入全局字典
    """
    # Load data only if the global dict is empty
    # 仅当全局字典为空时加载数据
    if not external_data:
        external_data_path = get_abs_path(agent_conf["external_data_path"])

        # Check if external data file exists
        # 检查外部数据文件是否存在
        if not os.path.exists(external_data_path):
            logger.error(f"[-][get_current_month] External file {external_data_path} not found")
            raise FileNotFoundError(f"External file {external_data_path} not found")

        # Open and read CSV line by line (skip header)
        # 打开并逐行读取CSV（跳过表头）
        with open(external_data_path, "r", encoding="utf-8") as f:
            for line in f.readlines()[1:]:
                arr: list[str] = line.strip().split(",")

                # Parse each field
                # 解析每个字段
                user_id: str = arr[0]
                house_info: str = arr[1].replace('"', "")
                metrics: str = arr[2].replace('"', "")
                comment: str = arr[3].replace('"', "")
                time: str = arr[4].replace('"', "")

                # Initialize user node if not exists
                # 如果用户不存在，初始化用户节点
                if user_id not in external_data:
                    external_data[user_id] = {}

                # Store monthly data under the user
                # 将月度数据存入对应用户
                external_data[user_id][time] = {
                    "house_info": house_info,
                    "metrics": metrics,
                    "comment": comment,
                }

中间件(middleware)

@wrap_tool_call
def monitor_tool(
        request: ToolCallRequest,
        handler: Callable[[ToolCallRequest], ToolMessage | Command],
) -> ToolMessage | Command:
    """
    Middleware to monitor and log tool execution status.
    Logs tool name, arguments, success/failure, and updates runtime context for report mode.

    Args:
        request: Encapsulated tool call request with parameters and runtime info
        handler: The actual tool execution function to wrap

    Returns:
        ToolMessage or Command: Result from the tool execution
    """
    logger.info(f"[+][monitor_tool] Executing tool: {request.tool_call['name']}")
    logger.info(f"[+][monitor_tool] Parameters: {request.tool_call['args']}")

    try:
        result = handler(request)
        logger.info(f"[+][monitor_tool] Tool {request.tool_call['name']} executed successfully")

        if request.tool_call['name'] == "fill_context_for_report":
            request.runtime.context["report"] = True

        return result

    except Exception as e:
        logger.error(f"[-][monitor_tool] Tool {request.tool_call['name']} failed: {str(e)}")
        raise e


@before_model
def log_before_model(
        state: AgentState,
        runtime: Runtime,
):
    """
    Logs information before invoking the LLM.
    Includes safety checks to avoid index errors when accessing message history.

    Args:
        state: Current state of the agent containing message history
        runtime: Runtime context with execution metadata

    Returns:
        AgentState: Updated agent state
    """
    logger.info("[+][log_before_model] Preparing to call the large language model")

    if state.get("messages") and len(state["messages"]) > 0:
        last_msg = state["messages"][-1]
        if hasattr(last_msg, 'content') and last_msg.content:
            logger.debug(f"[+][log_before_model] {type(last_msg).__name__} | {last_msg.content.strip()}")

    return state


@dynamic_prompt
def report_prompt_switch(request: ModelRequest):
    """
    Dynamically switches system prompts based on runtime context.
    Uses report-specific prompt if report mode is activated; otherwise uses default prompt.

    Args:
        request: Model request containing runtime context flags

    Returns:
        str: Selected system prompt for the LLM
    """
    is_report = request.runtime.context.get("report", False)
    if is_report:
        return load_report_prompts()

    return load_system_prompts()

class ReactAgent:
    """
    A ReAct-based intelligent agent that integrates LLM, tools, and middleware.
    It supports tool calling, dynamic prompt switching, and streaming response output.
    基于ReAct架构的智能体，集成大模型、工具集与中间件，支持工具调用、动态提示词切换与流式响应。
    """

    def __init__(self):
        """
        Initialize the ReAct agent with LLM, system prompt, tools, and middleware.
        初始化ReAct智能体，加载大模型、系统提示词、工具集与中间件。
        """
        # Create an agent instance using LangChain create_agent
        self.agent = create_agent(
            # Large language model for reasoning and answering
            model=chat_model,
            # Base system prompt for the agent
            system_prompt=load_system_prompts(),
            # List of available tools for the agent to use
            tools=[
                rag_summarize,
                get_weather,
                get_user_location,
                get_user_id,
                get_month,
                get_current_month,
                fetch_external_data,
                fill_context_for_report
            ],
            # Middleware for tool monitoring, logging, and dynamic prompt switching
            middleware=[
                monitor_tool,
                log_before_model,
                report_prompt_switch
            ]
        )

    def execute_stream(self, query: str):
        """
        Execute the agent query and return streaming response chunks.
        执行用户查询并以流式方式返回响应片段。

        Args:
            query: User input question or instruction / 用户输入的问题或指令

        Yields:
            Streaming response content chunk / 流式输出的响应内容片段
        """
        # Construct input data in the format required by the agent
        input_dict = {
            "messages": [
                {"role": "user", "content": query},
            ]
        }

        # Stream agent responses with context (report mode disabled by default)
        for chunk in self.agent.stream(input_dict, stream_mode="values", context={"report": False}):
            # Get the latest message from the state
            latest_message = chunk["messages"][-1]
            # Yield non-empty content
            if latest_message.content:
                yield latest_message.content.strip() + "\n"
