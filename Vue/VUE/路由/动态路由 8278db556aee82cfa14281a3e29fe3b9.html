动态路由

动态路由是指路径中包含动态参数的路由，用于匹配不同的资源或页面。它允许我们使用相同的组件来渲染不同的内容。

  • 特点
    • 路径参数：路由路径中包含可变部分（如 /user/123）
    • 组件复用：同一个组件可以渲染不同的数据
    • 参数获取：通过 $route.params或 props获取参数
    • 响应式更新：参数变化时组件会自动更新
  • 基本语法
    import Vue from 'vue'
import VueRouter from 'vue-router'

Vue.use(VueRouter)

const User = () => import('@/views/User.vue')
const Product = () => import('@/views/Product.vue')

const routes = [
  // 🔹 基本动态路由
  {
    path: '/user/:id',        // :id 是动态参数
    name: 'User',
    component: User
  },
  
  // 🔹 多个动态参数
  {
    path: '/category/:categoryId/product/:productId',
    name: 'ProductDetail',
    component: Product
  },
  
  // 🔹 可选参数
  {
    path: '/article/:id?',    // ? 表示可选参数
    name: 'Article',
    component: () => import('@/views/Article.vue')
  }
]

const router = new VueRouter({
  mode: 'history',
  routes
})

export default router
  • 参数获取方式
    1. 通过$route
    <template>
  <div class="user">
    <h1>用户详情页面</h1>
    <div class="user-info">
      <p><strong>用户ID：</strong> {{ $route.params.id }}</p>
      <p><strong>完整路径：</strong> {{ $route.path }}</p>
      <p><strong>查询参数：</strong> {{ $route.query }}</p>
      <p><strong>哈希值：</strong> {{ $route.hash }}</p>
    </div>
    
    <!-- 动态生成导航链接 -->
    <nav class="user-nav">
      <router-link :to="`/user/${userId}/profile`">资料</router-link>
      <router-link :to="`/user/${userId}/posts`">帖子</router-link>
    </nav>
  </div>
</template>

<script>
export default {
  name: 'User',
  data() {
    return {
      // 从路由参数初始化数据
      userId: this.$route.params.id
    }
  },
  watch: {
    // 监听路由参数变化
    '$route.params.id': {
      handler(newId) {
        console.log('用户ID变化:', newId)
        this.userId = newId
        this.loadUserData(newId)
      },
      immediate: true
    }
  },
  methods: {
    loadUserData(userId) {
      console.log('加载用户数据:', userId)
      // 实际项目中调用API
    }
  },
  
  // 路由生命周期钩子
  beforeRouteUpdate(to, from, next) {
    console.log('路由参数更新:', from.params.id, '→', to.params.id)
    // 参数变化时重新加载数据
    this.loadUserData(to.params.id)
    next()
  }
}
</script>
    1. 通过props（推荐）
    const routes = [
  {
    path: '/user/:id',
    name: 'User',
    component: User,
    props: true  // 将路由参数作为 props 传递
  },
  {
    path: '/product/:category/:id',
    name: 'Product',
    component: Product,
    props: route => ({
      // 自定义 props 映射
      productId: route.params.id,
      category: route.params.category,
      query: route.query,
      mode: route.query.mode || 'view'
    })
  }
]
    <template>
  <div class="user">
    <h1>用户详情 - {{ id }}</h1>
    <!-- 直接使用 props -->
    <p>用户ID: {{ id }}</p>
    <p>查询模式: {{ mode }}</p>
  </div>
</template>

<script>
export default {
  name: 'User',
  // 通过 props 接收参数
  props: {
    id: {
      type: String,
      required: true
    },
    mode: {
      type: String,
      default: 'view'
    }
  },
  watch: {
    // 监听 props 变化
    id(newId) {
      this.loadUserData(newId)
    }
  },
  methods: {
    loadUserData(userId) {
      console.log('通过props加载数据:', userId)
    }
  }
}
</script>
  • 高级参数匹配
    1. 参数约束（正则表达式）
    const routes = [
  // 只匹配数字ID
  {
    path: '/user/:id(\\d+)',        // 只匹配数字
    name: 'User',
    component: User
  },
  
  // 多段参数
  {
    path: '/files/:path*',          // * 匹配多段路径
    name: 'Files',
    component: Files
  },
  
  // 可选多段参数
  {
    path: '/docs/:section?/:page?', // 可选参数
    name: 'Docs',
    component: Documentation
  },
  
  // 复杂正则匹配
  {
    path: '/date/:year(\\d{4})/:month(\\d{2})/:day(\\d{2})',
    name: 'Date',
    component: DatePage
  }
]
    1. 路由优先级
    const routes = [
  // 静态路径优先匹配
  {
    path: '/user/me',        // 优先匹配 /user/me
    name: 'CurrentUser',
    component: CurrentUser
  },
  {
    path: '/user/:id',       // 然后匹配动态路径
    name: 'User',
    component: User
  },
  {
    path: '/user/*',          // 最后匹配通配符
    name: 'UserWildcard',
    component: UserWildcard
  }
]
