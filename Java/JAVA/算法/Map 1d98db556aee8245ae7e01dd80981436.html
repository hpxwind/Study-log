compute 方法

Map<String, Integer> map = new HashMap<>();

// 1.1 compute - 计算新值
map.put("A", 1);
map.compute("A", (key, oldValue) -> oldValue + 1);  // A=2
map.compute("B", (key, oldValue) -> oldValue == null ? 1 : oldValue + 1);  // B=1

// 1.2 computeIfAbsent - 不存在时计算
map.computeIfAbsent("C", key -> 10);  // C不存在，设为10
map.computeIfAbsent("C", key -> 20);  // C已存在，不修改

// 复杂示例
Map<String, List<String>> multiMap = new HashMap<>();
multiMap.computeIfAbsent("fruits", k -> new ArrayList<>()).add("apple");
multiMap.computeIfAbsent("fruits", k -> new ArrayList<>()).add("banana");

// 1.3 computeIfPresent - 存在时计算
map.put("D", 5);
map.computeIfPresent("D", (key, oldValue) -> oldValue * 2);  // D=10
map.computeIfPresent("E", (key, oldValue) -> oldValue + 1);  // E不存在，不操作

merge 方法

Map<String, Integer> map = new HashMap<>();
map.put("A", 1);
map.put("B", 2);

// 2.1 合并值
map.merge("A", 1, Integer::sum);  // A=2 (1+1)
map.merge("C", 1, Integer::sum);  // C不存在，设为1
map.merge("B", 5, (oldVal, newVal) -> oldVal - newVal);  // B=-3 (2-5)

// 2.2 单词计数示例
String text = "apple banana apple orange banana apple";
Map<String, Integer> wordCount = new HashMap<>();

for (String word : text.split(" ")) {
    wordCount.merge(word, 1, Integer::sum);
}
// 结果: {apple=3, banana=2, orange=1}

特殊Map实现：LinkedHashMap (保持顺序)

// 1.1 保持插入顺序
Map<String, Integer> insertionOrder = new LinkedHashMap<>();
insertionOrder.put("C", 3);
insertionOrder.put("A", 1);
insertionOrder.put("B", 2);
// 遍历顺序: C -> A -> B

// 1.2 LRU缓存实现
class LRUCache<K, V> extends LinkedHashMap<K, V> {
    private final int capacity;
    
    public LRUCache(int capacity) {
        super(capacity, 0.75f, true);  // accessOrder=true
        this.capacity = capacity;
    }
    
    @Override
    protected boolean removeEldestEntry(Map.Entry<K, V> eldest) {
        return size() > capacity;
    }
}

LRUCache<String, Integer> cache = new LRUCache<>(3);
cache.put("A", 1);
cache.put("B", 2);
cache.put("C", 3);
cache.get("A");  // 访问A，A移动到末尾
cache.put("D", 4);  // 移除最老的B
// 当前顺序: C -> A -> D

    Comparator.comparing(Student::getScore).reversed()

// 2.3 范围查询
TreeMap<Integer, String> scores = new TreeMap<>(Map.of(
    85, "Alice", 92, "Bob", 78, "Charlie", 95, "Diana"
));

// 获取第一个/最后一个
Map.Entry<Integer, String> first = scores.firstEntry();  // 78=Charlie
Map.Entry<Integer, String> last = scores.lastEntry();    // 95=Diana

// 小于/小于等于
Integer lowerKey = scores.lowerKey(90);    // 85 (<90的最大键)
Integer floorKey = scores.floorKey(85);    // 85 (≤85的最大键)

// 大于/大于等于
Integer higherKey = scores.higherKey(90);  // 92 (>90的最小键)
Integer ceilingKey = scores.ceilingKey(85); // 85 (≥85的最小键)

// 子Map
SortedMap<Integer, String> headMap = scores.headMap(90);      // 键<90
SortedMap<Integer, String> tailMap = scores.tailMap(85);      // 键≥85
SortedMap<Integer, String> subMap = scores.subMap(80, 95);    // 80≤键<95

分组和分区

List<String> words = Arrays.asList("apple", "banana", "cherry", "date", "elderberry");

// 2.1 按长度分组
Map<Integer, List<String>> byLength = words.stream()
    .collect(Collectors.groupingBy(String::length));

// 2.2 按条件分区
Map<Boolean, List<String>> partitioned = words.stream()
    .collect(Collectors.partitioningBy(w -> w.startsWith("a")));

// 2.3 复杂分组
Map<Integer, Long> countByLength = words.stream()
    .collect(Collectors.groupingBy(String::length, Collectors.counting()));

Map<Integer, Set<String>> setByLength = words.stream()
    .collect(Collectors.groupingBy(String::length, Collectors.toSet()));
