Ajax

ajax === async javascript and xml（异步的 JavaScript 和 XML）

  • node.js相关安装

    官网：https://nodejs.org/

    node -v
npm -v
npm install json-server -g # -g标识全局安装
  • ajax 状态码

    xhr.readyState

    readyState === 0: 未初始化完成，open还还未执行

    readyState === 1: 配置信息完成，open执行完之后

    readyState === 2: send方法执行完毕

    readyState === 3: 正在解析响应内容

    readyState === 4: 标识响应内容解析完毕，可以在客户端使用

    xhr.onreadystatechange事件会检测到 2，3，4种状态

    xhr.onload事件只会检测4状态

    xhr.status 为资源状态码 如：200，300，404，505

    xhr.statusText 输出状态码描述 如：OK

    xhr.responseText 接收返回的响应，输出为文本

  • 封装

    将ajax打包成一个方法，使用时候直接调用即可，打包如下有注释

    /**
 * 基础AJAX封装函数
 * @param {Object} options - 请求配置选项
 * @param {string} options.url - 请求的URL地址（必填）
 * @param {string} [options.method='GET'] - 请求方法，如'GET'、'POST'、'PUT'等
 * @param {Object} [options.data] - 发送的数据（POST/PUT等方法使用）
 * @param {Object} [options.params] - URL查询参数（GET方法使用）
 * @param {Object} [options.headers] - 请求头设置
 * @param {number} [options.timeout=5000] - 超时时间（毫秒）
 * @returns {Promise} - 返回Promise对象，成功时返回响应数据，失败时返回错误信息
 */
function ajax(options) {
    // 返回Promise，方便使用then/catch和async/await
    return new Promise((resolve, reject) => {
        // 1. 处理默认参数
        const {
            url,
            method = 'GET',
            data = null,
            params = null,
            headers = {},
            timeout = 5000
        } = options;

        // 2. 处理URL和查询参数（GET请求的参数拼接）
        let fullUrl = url;
        if (params) {
            // 将参数对象转为查询字符串（key=value&key2=value2）
            const paramsArray = [];
            for (const key in params) {
                // 编码特殊字符，避免URL解析错误
                paramsArray.push(`${encodeURIComponent(key)}=${encodeURIComponent(params[key])}`);
            }
            const paramsString = paramsArray.join('&');
            // 处理URL中已有?的情况
            fullUrl += (fullUrl.includes('?') ? '&' : '?') + paramsString;
        }

        // 3. 创建XMLHttpRequest对象
        const xhr = new XMLHttpRequest();

        // 4. 初始化请求（method, url, async）
        xhr.open(method.toUpperCase(), fullUrl, true);

        // 5. 设置请求头
        // 默认设置Content-Type，用户传入的headers会覆盖默认值
        const defaultHeaders = {
            'Content-Type': 'application/json'
        };
        const requestHeaders = { ...defaultHeaders, ...headers };
        for (const key in requestHeaders) {
            xhr.setRequestHeader(key, requestHeaders[key]);
        }

        // 6. 设置超时处理
        xhr.timeout = timeout;
        xhr.ontimeout = () => {
            reject(new Error(`请求超时（${timeout}ms）`));
        };

        // 7. 处理响应
        // HTTP状态码错误（如404, 500）- 仍在 onload 中处理
        // 如：服务器返回404，这是HTTP层面的"成功"响应，但内容不存在
        xhr.onload = () => {
            // 状态码在200-299之间表示请求成功
            if (/^2\d{2}$/.test(xhr.status)) {
                try {
                    // 尝试解析JSON响应
                    const responseData = JSON.parse(xhr.responseText);
                    resolve(responseData);
                } catch (e) {
                    // 如果不是JSON格式，直接返回文本
                    resolve(xhr.responseText);
                }
            } else {
                // 非成功状态码，返回错误信息
                reject(new Error(`请求失败，状态码：${xhr.status}，信息：${xhr.statusText}`));
            }
        };

        // 8. 处理网络错误
        // 当网络层面错误发生时，浏览器会自动调用这个函数
        xhr.onerror = () => {
            reject(new Error('网络错误，请检查网络连接'));
        };

        // 9. 处理请求数据
        let sendData = null;
        if (data) {
            const contentType = requestHeaders['Content-Type'];
            if (contentType.includes('application/json')) {
                // JSON格式：将对象转为JSON字符串
                sendData = JSON.stringify(data);
            } else if (contentType.includes('application/x-www-form-urlencoded')) {
                // 表单格式：转为key=value&key2=value2
                const dataArray = [];
                for (const key in data) {
                    dataArray.push(`${encodeURIComponent(key)}=${encodeURIComponent(data[key])}`);
                }
                sendData = dataArray.join('&');
            } else {
                // 其他格式（如FormData）直接发送
                sendData = data;
            }
        }

        // 10. 发送请求
        xhr.send(sendData);
    });
}
