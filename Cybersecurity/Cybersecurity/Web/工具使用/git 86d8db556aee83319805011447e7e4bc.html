git

  • 基础配置
    # 1. 设置用户名（与 GitHub/GitLab 账号一致）
git config --global user.name "Your Name"

# 2. 设置邮箱（与账号绑定的邮箱）
git config --global user.email "your.email@example.com"

# 3. 查看配置信息
git config --list

# 4. 配置默认编辑器（如 VS Code）
git config --global core.editor "code --wait"
  • 仓库初始化与克隆
    # 1. 初始化本地仓库（在项目目录执行）
git init

# 2. 克隆远程仓库到本地（复制项目）
git clone <远程仓库地址>  # 例如：git clone https://github.com/username/repo.git

# 3. 克隆到指定目录
git clone <远程仓库地址> <本地目录名>
  • 文件状态与暂存

    git 中文件有三种状态：工作区（Working Directory）→ 暂存区（Stage/Index） → 本地仓库（Repository）

    # 1. 查看文件状态
git status  # 详细显示状态
git status -s  # 简洁显示（??: 未跟踪；A: 已暂存；M: 已修改）

# 2. 将工作区文件添加到暂存区
git add <文件名>  # 添加单个文件
git add <目录名>  # 添加整个目录
git add .  # 添加当前目录所有文件（不包括被忽略的）
git add -A  # 添加所有变化（包括删除的文件）

# 3. 从暂存区移除文件（保留工作区文件）
git reset <文件名>

# 4. 放弃工作区修改（恢复到最近一次提交状态）
git checkout -- <文件名>  # 谨慎使用！会丢失未提交的修改

# 5. 忽略文件（创建 .gitignore 文件）
echo "node_modules/" >> .gitignore  # 忽略 node_modules 目录
echo "*.log" >> .gitignore  # 忽略所有 .log 文件
  • 提交与分支操作
    # 1. 提交暂存区文件到本地仓库
git commit -m "提交说明"  # 说明需简洁明了（如："fix: 修复登录bug"）

# 2. 直接提交工作区已跟踪的文件（跳过暂存区）
git commit -am "提交说明"  # 仅适用于已跟踪的文件（已 add 过的）

# 3. 修改最近一次提交（未推送到远程时使用）
git commit --amend  # 会打开编辑器修改提交信息
git commit --amend -m "新的提交说明"  # 直接修改说明

    分支用于并行开发

    # 1. 提交暂存区文件到本地仓库
git commit -m "提交说明"  # 说明需简洁明了（如："fix: 修复登录bug"）

# 2. 直接提交工作区已跟踪的文件（跳过暂存区）
git commit -am "提交说明"  # 仅适用于已跟踪的文件（已 add 过的）

# 3. 修改最近一次提交（未推送到远程时使用）
git commit --amend  # 会打开编辑器修改提交信息
git commit --amend -m "新的提交说明"  # 直接修改说明

    合并分支

    # 1. 切换到目标分支（如主分支）
git checkout main

# 2. 合并来源分支（如 feature 分支）
git merge <来源分支名>  # 例如：git merge feature/login

# 3. 处理合并冲突
# 冲突时文件中会显示 <<<<<<< HEAD（当前分支）和 >>>>>>> 分支名（来源分支）
# 手动编辑文件解决冲突后，执行：
git add <冲突文件>
git commit -m "解决合并冲突"

# 4. 取消合并（合并冲突时）
git merge --abort

  • 远程仓库操作
    # 1. 查看远程仓库信息
git remote  # 显示远程仓库别名（默认 origin）
git remote -v  # 显示远程仓库地址

# 2. 添加远程仓库
git remote add <别名> <远程地址>  # 例如：git remote add origin https://xxx.git

# 3. 从远程仓库拉取最新代码（不合并）
git fetch <远程别名>  # 例如：git fetch origin

# 4. 拉取并合并远程代码（常用，等价于 fetch + merge）
git pull <远程别名> <远程分支>  # 例如：git pull origin main

# 5. 推送本地分支到远程仓库
git push <远程别名> <本地分支>  # 例如：git push origin main

# 6. 推送新分支到远程（首次推送）
git push -u <远程别名> <本地分支>  # -u 关联远程分支，后续可直接 git push

# 7. 删除远程分支
git push <远程别名> --delete <远程分支名>  # 例如：git push origin --delete feature/test

# 8. 更换远程仓库地址
git remote set-url origin <新地址>
  • 退回操作
    # 1. 查看提交历史（详细）
git log  # 按 q 退出
git log --oneline  # 简洁显示（一行一个提交）
git log --graph  # 图形化显示分支合并历史

# 2. 回退到指定版本（本地仓库）
git reset --hard <提交ID>  # 提交ID可通过 git log 查看前几位

# 3. 回退到上一个版本
git reset --hard HEAD~1  # ~n 表示回退 n 个版本

# 4. 撤销已推送到远程的提交（谨慎！会覆盖远程历史）
git revert <提交ID>  # 创建一个新提交来抵消指定提交的修改
  • 查看文件内容操作
    查看提交日志：
git log：显示当前分支的提交历史，包括提交哈希值、作者、日期和提交信息等。按q键可退出查看。
git log -n：n为具体数字，用于显示最近n次提交。例如git log -5会显示最近 5 次提交记录。
git log --stat：显示 commit 历史，以及每次 commit 发生变更的文件，会列出文件的增删行数等统计信息。
git log --oneline：单行显示提交记录，只显示简短的 commit id 及提交日志。
查看文件差异：
git diff：查看工作区与暂存区之间的差异，即显示已修改但未暂存的文件内容变化。
git diff --cached：查看暂存区与本地仓库中最新提交之间的差异，也就是已暂存但未提交的文件变化。
git diff HEAD：显示工作区与当前分支最新 commit 之间的差异。
git diff <first - branch> <second - branch>：显示两个分支之间的差异。
查看某次提交的内容：
git show <commit>：显示指定 commit 的元数据和内容变化，包括提交者、日期、提交信息以及文件的改动等。
git show --name - only <commit>：仅显示指定 commit 中发生变化的文件名称。
git show <commit>:<filename>：显示指定 commit 时，某个具体文件的内容。例如git show abc1234:README.md，会显示哈希值为abc1234的提交中README.md文件的内容。
查看文件相关信息：
git status：查看工作区中文件的状态，包括哪些是未跟踪文件、哪些文件已修改但未暂存、哪些已暂存等。
git blame <file>：显示指定文件的每一行最后是由谁在何时修改的，常用于追溯代码的修改历史。
git whatchanged <file>：显示某个文件的版本历史，包括文件改名等情况。
