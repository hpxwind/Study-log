网络编程

TCP协议

tcp面向连接的运输层协议，比较复杂，安全，效率低，使用tcp协议前需要建立连接，才能传输数据，数据传输完毕之后需要释放连接

ServerScoket

方法描述
public ServerSocket(int port)根据端口创建ServerSocket对象
public ServerSocket(int port,int backlog,InetAddress address)根据端口，backlog，IP地址创建对象
public Socket accept()等待客户端请求，返回Socket对象
public void close()关闭ServerSocket

Socket

方法描述
public Socket(String host, int port)根据主机，端口创建要连接的Socket对象
public Socket(InetAddress host,int port)根据ip，端口创建要连接的Socket对象
public InputStream getInputStream()获取Socket输入流
public OutputStream getOutPutSream()获取Socket输出流（发送信息）
public synchronized void close()关闭Socket
  • 简单例子

    Server.java

    import java.io.*;
import java.net.ServerSocket;
import java.net.Socket;

public class Server {

    public static void main(String[] args) throws IOException {
        // 1. 创建ServerSocket对象，绑定8888端口
        try (ServerSocket serverSocket = new ServerSocket(8888)) {
            System.out.println("服务器已连接，等待客户端连接...");

            // 2. 阻塞等待客户端连接，返回Socket对象
            Socket socket = serverSocket.accept();
            System.out.println("客户端已连接: " + socket.getInetAddress());

            // 3. 获取Socket 输入流，读取客户端发送的数据
            InputStream inputStream = socket.getInputStream();
            BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
            String message = reader.readLine();
            System.out.println("收到: " + message);

            // 4. 回复
            String resp = "Hello world";
            OutputStream outputStream = socket.getOutputStream();
            outputStream.write((resp+"\n").getBytes());

            // 5. 关闭socket连接
            socket.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

UDP协议

UDP协议相反，效率高，不安全

DatagramSocket

方法描述
public DatagramSocket(int port)根据端口创建DatagramSocket对象
public void send(DatagramPacket p)发送数据包
public synchronized receive(DatagramPacket p)接收数据包

DatagramPacket

方法描述
public DatagramPacket(byte buf[],int length, IntAddress address, int port)根据发送的数据，数据长度，IP地址，端口创建DatagramPacket对象
public synchronized byte[] getData()获取接收的数据
public synchronized int getLength()获取数据长度
public sunchronized int getPort()获取发送数据的Socket 端口
  • 简单案例

    Server.java

    import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;

public class UDPReceiver {
    public static void main(String[] args) {
        // 1. 根据端口创建 DatagramSocket
        try (DatagramSocket socket = new DatagramSocket(9999)) {
            System.out.println("UDP 接收端已启动，等待数据...");

            // 准备缓冲区
            byte[] buf = new byte[1024];
            DatagramPacket packet = new DatagramPacket(buf, buf.length);

            // 2. 阻塞接收数据包
            socket.receive(packet);

            // 3. 获取数据、长度、端口
            byte[] data = packet.getData();         // 获取数据
            int len = packet.getLength();          // 获取数据长度
            int port = packet.getPort();            // 获取对方端口
            InetAddress ip = packet.getAddress();  // 获取对方IP

            String msg = new String(data, 0, len);
            System.out.println("收到来自 " + ip + ":" + port + " 的消息：" + msg);

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

    Client.java

    import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;

public class UDPSender {
    public static void main(String[] args) {
        // 1. 创建发送用的 DatagramSocket（端口自动分配）
        try (DatagramSocket socket = new DatagramSocket()) {
            // 目标地址和端口
            InetAddress ip = InetAddress.getByName("127.0.0.1");
            int port = 9999;
            String msg = "Hello UDP！";

            // 2. 创建数据包：数据、长度、IP、端口
            DatagramPacket packet = new DatagramPacket(
                    msg.getBytes(),
                    msg.getBytes().length,
                    ip,
                    port
            );

            // 3. 发送数据包
            socket.send(packet);
            System.out.println("发送完成！");

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

多线程下的网络编程

每来一个客户端就创建一个新的线程处理，主线程负责监听连接，不处理业务

  • 例子

    Server.java

    import java.io.*;
import java.net.*;

public class MultiThreadServer {
    public static void main(String[] args) throws IOException {
        // 监听 8888 端口
        ServerSocket serverSocket = new ServerSocket(8888);
        System.out.println("多线程服务端已启动，等待客户端连接...");

        while (true) {
            // 阻塞等待客户端连接
            Socket socket = serverSocket.accept();
            System.out.println("新客户端连接：" + socket.getInetAddress());

            // 为每个客户端创建新线程
            new Thread(new ClientHandler(socket)).start();
        }
    }
}

// 客户端处理线程（核心）
class ClientHandler implements Runnable {
    private Socket socket;

    public ClientHandler(Socket socket) {
        this.socket = socket;
    }

    @Override
    public void run() {
        try (
            BufferedReader in = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            PrintWriter out = new PrintWriter(socket.getOutputStream(), true)
        ) {
            String msg;
            // 循环读取客户端消息
            while ((msg = in.readLine()) != null) {
                System.out.println("收到客户端：" + msg);
                out.println("服务端已收到：" + msg); // 回写数据
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                socket.close();
            } catch (IOException e) {}
        }
    }
}

    Client.java

    import java.io.*;
import java.net.*;
import java.util.Scanner;

public class Client {
    public static void main(String[] args) throws IOException {
        Socket socket = new Socket("127.0.0.1", 8888);
        Scanner scanner = new Scanner(System.in);

        BufferedReader in = new BufferedReader(new InputStreamReader(socket.getInputStream()));
        PrintWriter out = new PrintWriter(socket.getOutputStream(), true);

        while (true) {
            System.out.print("请输入消息：");
            String msg = scanner.nextLine();
            out.println(msg);

            String resp = in.readLine();
            System.out.println(resp);
        }
    }
}
